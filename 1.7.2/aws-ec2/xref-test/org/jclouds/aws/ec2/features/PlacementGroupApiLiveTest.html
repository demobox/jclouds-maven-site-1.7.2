88     @Test
89     public void testDescribe() {
90        for (String region : supportedRegions) {
91           SortedSet<PlacementGroup> allResults = newTreeSet(client.getPlacementGroupApi().get()
92                    .describePlacementGroupsInRegion(region));
93           assertNotNull(allResults);
94           if (allResults.size() >= 1) {
95              PlacementGroup group = allResults.last();
96              SortedSet<PlacementGroup> result = newTreeSet(client.getPlacementGroupApi().get()
97                       .describePlacementGroupsInRegion(region, group.getName()));
98              assertNotNull(result);
99              PlacementGroup compare = result.last();
100             assertEquals(compare, group);
101          }
102       }
103 
104       for (String region : client.getAvailabilityZoneAndRegionApi().get().describeRegions().keySet()) {
105          if (!supportedRegions.contains(region))
106             try {
107                client.getPlacementGroupApi().get().describePlacementGroupsInRegion(region);
108                fail("should be unsupported for region: " + region);
109             } catch (UnsupportedOperationException e) {
110             }
111       }
112    }

149    @Test
150    public void testCreatePlacementGroup() {
151       String groupName = PREFIX + "1";
152       for (String region : supportedRegions) {
153 
154          client.getPlacementGroupApi().get().deletePlacementGroupInRegion(region, groupName);
155          client.getPlacementGroupApi().get().createPlacementGroupInRegion(region, groupName);
156 
157          verifyPlacementGroup(region, groupName);
158       }
159    }

161    private void verifyPlacementGroup(String region, String groupName) {
162       assert availableTester.apply(new PlacementGroup(region, groupName, "cluster", State.PENDING)) : group;
163       Set<PlacementGroup> oneResult = client.getPlacementGroupApi().get().describePlacementGroupsInRegion(region,
164                groupName);
165       assertNotNull(oneResult);
166       assertEquals(oneResult.size(), 1);
167       group = oneResult.iterator().next();
168       assertEquals(group.getName(), groupName);
169       assertEquals(group.getStrategy(), "cluster");
170       assert availableTester.apply(group) : group;
171    }

173    public void testStartCCInstance() throws Exception {
174 
175       Template template = view.getComputeService().templateBuilder()
176                .fromHardware(EC2HardwareBuilder.cc2_8xlarge().build()).osFamily(OsFamily.AMZN_LINUX).build();
177       assert template != null : "The returned template was null, but it should have a value.";
178       assertEquals(template.getHardware().getProviderId(), InstanceType.CC2_8XLARGE);
179       assertEquals(template.getImage().getUserMetadata().get("rootDeviceType"), "ebs");
180       assertEquals(template.getImage().getUserMetadata().get("virtualizationType"), "hvm");
181       assertEquals(template.getImage().getUserMetadata().get("hypervisor"), "xen");
182 
183       template.getOptions().runScript(Statements.newStatementList(AdminAccess.standard(), InstallJDK.fromOpenJDK()));
184 
185       String group = PREFIX + "cccluster";
186       view.getComputeService().destroyNodesMatching(NodePredicates.inGroup(group));
187       // TODO make this not lookup an explicit region
188       client.getPlacementGroupApi().get().deletePlacementGroupInRegion(null, "jclouds#" + group + "#us-east-1");
189 
190       try {
191          Set<? extends NodeMetadata> nodes = view.getComputeService().createNodesInGroup(group, 1, template);
192          NodeMetadata node = getOnlyElement(nodes);
193 
194          getOnlyElement(getOnlyElement(client.getInstanceApi().get().describeInstancesInRegion(null,
195                   node.getProviderId())));
196 
197       } catch (RunNodesException e) {
198          System.err.println(e.getNodeErrors().keySet());
199          Throwables.propagate(e);
200       } finally {
201          view.getComputeService().destroyNodesMatching(NodePredicates.inGroup(group));
202       }
203    }
