43  @RequestFilters(FormSigner.class)
44  @VirtualHost
45  public interface AWSKeyPairApi extends KeyPairApi {
46  
47     /**
48      * Imports the public key from an RSA key pair that you created with a third-party tool. Compare
49      * this with CreateKeyPair, in which AWS creates the key pair and gives the keys to you (AWS
50      * keeps a copy of the public key). With ImportKeyPair, you create the key pair and give AWS just
51      * the public key. The private key is never transferred between you and AWS.
52      * 
53      * <p/>
54      * You can easily create an RSA key pair on Windows and Linux using the ssh-keygen command line
55      * tool (provided with the standard OpenSSH installation). Standard library support for RSA key
56      * pair creation is also available in Java, Ruby, Python, and many other programming languages.
57      * 
58      * <p/>
59      * <h4>Supported Formats</h4>
60      * <ul>
61      * <li>OpenSSH public key format (e.g., the format in ~/.ssh/authorized_keys)</li>
62      * <li>Base64 encoded DER format</li>
63      * <li>SSH public key file format as specified in RFC4716</li>
64      * </ul>
65      * DSA keys are not supported. Make sure your key generator is set up to create RSA keys.
66      * <p/>
67      * Supported lengths: 1024, 2048, and 4096.
68      * <p/>
69      * 
70      * @param region
71      *           region to import the key into
72      * @param keyName
73      *           A unique name for the key pair. Accepts alphanumeric characters, spaces, dashes, and
74      *           underscores.
75      * @param publicKeyMaterial
76      *           The public key
77      * @return imported key including fingerprint
78      */
79     @Named("ImportKeyPair")
80     @POST
81     @Path("/")
82     @FormParams(keys = ACTION, values = "ImportKeyPair")
83     @XMLResponseParser(KeyPairResponseHandler.class)
84     KeyPair importKeyPairInRegion(
85           @EndpointParam(parser = RegionToEndpointOrProviderIfNull.class) @Nullable String region,
86           @FormParam("KeyName") String keyName,
87           @FormParam("PublicKeyMaterial") @ParamParser(EncodedRSAPublicKeyToBase64.class) String publicKeyMaterial);
88  }
