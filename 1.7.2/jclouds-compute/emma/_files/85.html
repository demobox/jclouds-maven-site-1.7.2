485   static final Ordering<Image> DEFAULT_IMAGE_ORDERING = new Ordering<Image>() {
486      public int compare(Image left, Image right) {
487         /* This currently, and for some time, has *preferred* images whose fields are null,
488          * and prefers those which come last alphabetically.
489          * It seems preferable to take images whose fields are *not* null, ie nullsFirst;
490          * and to use something like the AlphaNum Algorithm then take the last
491          * (so "Ubuntu 13.04" would be preferred over "Ubuntu 9.10").
492          * However not changing it now as people may be surprised if the images they get back start changing.
493          */
494         return ComparisonChain.start()
495               .compare(left.getName(), right.getName(), Ordering.<String> natural().nullsLast())
496               .compare(left.getVersion(), right.getVersion(), Ordering.<String> natural().nullsLast())
497               .compare(left.getDescription(), right.getDescription(), Ordering.<String> natural().nullsLast())
498               .compare(left.getOperatingSystem().getName(), right.getOperatingSystem().getName(),//
499                     Ordering.<String> natural().nullsLast())
500               .compare(left.getOperatingSystem().getVersion(), right.getOperatingSystem().getVersion(),//
501                     Ordering.<String> natural().nullsLast())
502               .compare(left.getOperatingSystem().getDescription(), right.getOperatingSystem().getDescription(),//
503                     Ordering.<String> natural().nullsLast())
504               .compare(left.getOperatingSystem().getArch(), right.getOperatingSystem().getArch(),//

506      }

844   /**
845    * Like Ordering, but handle the case where there are multiple valid maximums
846    */
847   @SuppressWarnings("unchecked")
848   @VisibleForTesting
849   static <T, E extends T> List<E> multiMax(Comparator<T> ordering, Iterable<E> iterable) {
850      Iterator<E> iterator = iterable.iterator();
851      List<E> maxes = newArrayList(iterator.next());
852      E maxSoFar = maxes.get(0);
853      while (iterator.hasNext()) {
854         E current = iterator.next();
855         int comparison = ordering.compare(maxSoFar, current);
856         if (comparison == 0) {
857            maxes.add(current);
858         } else if (comparison < 0) {
859            maxes = newArrayList(current);
860            maxSoFar = current;
861         }
862      }
863      return maxes;
864   }
