51  @Test(singleThreaded = true)
52  public class SequentialMultipartUploadStrategyMockTest {
53  
54     String authRequestBody = KeystoneFixture.INSTANCE.initialAuthWithUsernameAndPassword("user", "password")
55           .getPayload().getRawContent().toString();
56     String authResponse = KeystoneFixture.INSTANCE.responseWithAccess().getPayload().getRawContent().toString()
57           .replace("https://objects.jclouds.org/v1.0/40806637803162", "URL");
58     String token = "Auth_4f173437e4b013bee56d1007";
59  
60     @Test
61     public void testMPUDoesMultipart() throws IOException, InterruptedException {
62        MockWebServer server = new MockWebServer();
63        AtomicReference<URL> url = setURLReplacingDispatcher(server);
64        server.enqueue(new MockResponse().setResponseCode(200).setBody(authResponse));
65        server.enqueue(new MockResponse().setResponseCode(200).addHeader("ETag", "a00"));
66        server.enqueue(new MockResponse().setResponseCode(200).addHeader("ETag", "b00"));
67        server.enqueue(new MockResponse().setResponseCode(200).addHeader("ETag", "fff"));
68        server.play();
69        url.set(server.getUrl("/"));
70  
71        byte[] bytes = "0123456789abcdef".getBytes(Charsets.US_ASCII);
72        int partSize = bytes.length / 2;
73        SequentialMultipartUploadStrategy api = mockSequentialMultipartUploadStrategy(url.get().toString(), partSize);
74  
75        try {
76           assertEquals(api.execute("container", new BlobBuilderImpl().name("foo").payload(bytes).build()), "fff");
77        } finally {
78           RecordedRequest authRequest = server.takeRequest();
79           assertEquals(authRequest.getRequestLine(), "POST /tokens HTTP/1.1");
80           assertEquals(new String(authRequest.getBody()), authRequestBody);
81           assertEquals(authRequest.getHeader("Content-Length"), String.valueOf(authRequestBody.length()));
82  
83           RecordedRequest part1 = server.takeRequest();
84           assertEquals(part1.getRequestLine(), "PUT /container/foo/1 HTTP/1.1");
85           assertEquals(part1.getHeader("X-Auth-Token"), token);
86           assertEquals(part1.getHeader("Content-Length"), String.valueOf(partSize));
87           assertEquals(new String(part1.getBody()), "01234567");
88  
89           RecordedRequest part2 = server.takeRequest();
90           assertEquals(part2.getRequestLine(), "PUT /container/foo/2 HTTP/1.1");
91           assertEquals(part2.getHeader("X-Auth-Token"), token);
92           assertEquals(part2.getHeader("Content-Length"), String.valueOf(partSize));
93           assertEquals(new String(part2.getBody()), "89abcdef");
94  
95           RecordedRequest manifest = server.takeRequest();
96           assertEquals(manifest.getRequestLine(), "PUT /container/foo HTTP/1.1");
97           assertEquals(manifest.getHeader("X-Auth-Token"), token);
98           assertEquals(manifest.getHeader("Content-Length"), "0");
99  
100          server.shutdown();
101       }
102    }

104    private static final Set<Module> modules = ImmutableSet.<Module> of(new ExecutorServiceModule(sameThreadExecutor(),
105          sameThreadExecutor()));

107    static SequentialMultipartUploadStrategy mockSequentialMultipartUploadStrategy(String uri, int partSize) {
108       Properties overrides = new Properties();
109       // prevent expect-100 bug http://code.google.com/p/mockwebserver/issues/detail?id=6
110       overrides.setProperty(PROPERTY_SO_TIMEOUT, "0");
111       overrides.setProperty(PROPERTY_MAX_RETRIES, "1");
112       overrides.setProperty("jclouds.mpu.parts.size", String.valueOf(partSize));
113       return ContextBuilder.newBuilder("swift-keystone")
114                            .credentials("user", "password").endpoint(uri)
115                            .overrides(overrides)
116                            .modules(modules)
117                            .buildInjector().getInstance(SequentialMultipartUploadStrategy.class);
118    }

120    /**
121     * there's no built-in way to defer evaluation of a response header, hence
122     * this method, which allows us to send back links to the mock server.
123     */
124    private AtomicReference<URL> setURLReplacingDispatcher(MockWebServer server) {
125       final AtomicReference<URL> url = Atomics.newReference();
126 
127       final QueueDispatcher dispatcher = new QueueDispatcher() {
128          protected final BlockingQueue<MockResponse> responseQueue = new LinkedBlockingQueue<MockResponse>();
129 
130          @Override
131          public MockResponse dispatch(RecordedRequest request) throws InterruptedException {
132             MockResponse response = responseQueue.take();
133             if (response.getBody() != null) {
134                String newBody = new String(response.getBody()).replace("URL", url.get().toString());
135                response = response.setBody(newBody);
136             }
137             return response;
138          }
139 
140          @Override
141          public void enqueueResponse(MockResponse response) {
142             responseQueue.add(response);
143          }
144       };
145       server.setDispatcher(dispatcher);
146       return url;
147    }
