122             ).getUserApiForZoneAndInstance("RegionOne", "instanceId-1234-5678");
123 
124       Set<String> databases1 = Sets.newHashSet();
125       databases1.add( "databaseA" );

135       Set<User> users = Sets.newHashSet();
136       users.add(user1);
137       users.add(user2);
138       users.add(user3);
139
