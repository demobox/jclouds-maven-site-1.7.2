61  /**
62   * Signer for HP's variant of temporary signed URLs. They prefix the signature with the tenant id.
63   * 
64   * @author Andrew Gaul
65   */
66  @Singleton
67  public class HPCloudObjectStorageBlobRequestSigner implements BlobRequestSigner {
68  
69     private final Function<Invocation, HttpRequest> processor;
70     private final Crypto crypto;
71  
72     private final Provider<Long> unixEpochTimestampProvider;
73     private final Supplier<Access> access;
74     private final Supplier<Credentials> creds;
75  
76     private final BlobToObject blobToObject;
77     private final BlobToHttpGetOptions blob2HttpGetOptions;
78  
79     private final Invokable<?, ?> getMethod;
80     private final Invokable<?, ?> deleteMethod;
81     private final Invokable<?, ?> createMethod;
82  
83     @Inject
84     public HPCloudObjectStorageBlobRequestSigner(Function<Invocation, HttpRequest> processor,
85           BlobToObject blobToObject, BlobToHttpGetOptions blob2HttpGetOptions, Crypto crypto,
86           @TimeStamp Provider<Long> unixEpochTimestampProvider, Supplier<Access> access,
87           @org.jclouds.location.Provider final Supplier<Credentials> creds) throws SecurityException,
88           NoSuchMethodException {
89        this.processor = checkNotNull(processor, "processor");
90        this.crypto = checkNotNull(crypto, "crypto");
91  
92        this.unixEpochTimestampProvider = checkNotNull(unixEpochTimestampProvider, "unixEpochTimestampProvider");
93        this.access = checkNotNull(access, "access");
94        this.creds = checkNotNull(creds, "creds");
95  
96        this.blobToObject = checkNotNull(blobToObject, "blobToObject");
97        this.blob2HttpGetOptions = checkNotNull(blob2HttpGetOptions, "blob2HttpGetOptions");
98  
99        this.getMethod = method(HPCloudObjectStorageAsyncApi.class, "getObject", String.class, String.class, GetOptions[].class);
100       this.deleteMethod = method(HPCloudObjectStorageAsyncApi.class, "removeObject", String.class, String.class);
101       this.createMethod = method(HPCloudObjectStorageAsyncApi.class, "putObject", String.class, SwiftObject.class);
102    }

104    @Override
105    public HttpRequest signGetBlob(String container, String name) {
106       checkNotNull(container, "container");
107       checkNotNull(name, "name");
108       return cleanRequest(processor.apply(Invocation.create(getMethod, ImmutableList.<Object> of(container, name))));
109    }

111    @Override
112    public HttpRequest signGetBlob(String container, String name, long timeInSeconds) {
113       checkNotNull(container, "container");
114       checkNotNull(name, "name");
115       HttpRequest request = processor.apply(Invocation.create(getMethod, ImmutableList.<Object> of(container, name)));
116       return cleanRequest(signForTemporaryAccess(request, timeInSeconds));

151    private HttpRequest signForTemporaryAccess(HttpRequest request, long timeInSeconds) {
152       Credentials currentCreds = checkNotNull(creds.get(), "credential supplier returned null");
153       // accessKey is of the form tenantName:accessKeyId (not tenantId)
154       String accessKeyId = currentCreds.identity.substring(currentCreds.identity.indexOf(':') + 1);
155       String secretKey = currentCreds.credential;
156       String tenantId = access.get().getToken().getTenant().get().getId();
157       
158       HttpRequest.Builder<?> builder = request.toBuilder();
159       // HP Cloud does not use X-Auth-Token for temporary signed URLs and
160       // leaking this allows clients arbitrary privileges until token timeout.
161       builder.filters(filter(request.getFilters(), not(instanceOf(AuthenticateRequest.class))));
162 
163       long expiresInSeconds = unixEpochTimestampProvider.get() + timeInSeconds;
164       String signature = createSignature(secretKey,
165             createStringToSign(request.getMethod().toUpperCase(), request, expiresInSeconds));
166 
167       builder.addQueryParam("temp_url_sig", String.format("%s:%s:%s", tenantId, accessKeyId, signature));
168       builder.addQueryParam("temp_url_expires", "" + expiresInSeconds);
169 
170       return builder.build();
171    }
172 
173    private String createStringToSign(String method, HttpRequest request, long expiresInSeconds) {
174       checkArgument(method.equalsIgnoreCase("GET") || method.equalsIgnoreCase("PUT"));
175       return String.format("%s\n%d\n%s", method.toUpperCase(), expiresInSeconds, request.getEndpoint().getPath());
176    }
177 
178    private String createSignature(String key, String toSign) {
179       try {
180          ByteProcessor<byte[]> hmacSHA1 = asByteProcessor(crypto.hmacSHA1(key.getBytes(UTF_8)));
181          return base16().lowerCase().encode(readBytes(toInputStream(toSign), hmacSHA1));
182       } catch (InvalidKeyException e) {
183          throw Throwables.propagate(e);
184       } catch (IOException e) {
185          throw Throwables.propagate(e);
186       }
187    }
