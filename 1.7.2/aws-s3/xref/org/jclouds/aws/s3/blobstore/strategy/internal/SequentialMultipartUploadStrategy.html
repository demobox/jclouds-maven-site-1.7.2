41  /**
42   * Provides a sequential multipart upload strategy.
43   * 
44   * The file partitioning algorithm:
45   * 
46   * The default partition size we choose is 32mb. A multiple of this default
47   * partition size is used. The number of partCount first grows to a chosen magnitude
48   * (for example 100 partCount), then it grows the partition size instead of number
49   * of partitions. When we reached the maximum part size, then again it starts to
50   * grow the number of partitions.
51   * 
52   * @author Tibor Kiss, Adrian Cole
53   */
54  public class SequentialMultipartUploadStrategy implements MultipartUploadStrategy {
55     @Resource
56     @Named(BlobStoreConstants.BLOBSTORE_LOGGER)
57     private Logger logger = Logger.NULL;
58  
59     private final AWSS3Client client;
60     private final BlobToObject blobToObject;
61     private final MultipartUploadSlicingAlgorithm algorithm;
62     private final PayloadSlicer slicer;
63  
64     @Inject
65     public SequentialMultipartUploadStrategy(AWSS3Client client, BlobToObject blobToObject,
66           MultipartUploadSlicingAlgorithm algorithm, PayloadSlicer slicer) {
67        this.client = checkNotNull(client, "client");
68        this.blobToObject = checkNotNull(blobToObject, "blobToObject");
69        this.algorithm = checkNotNull(algorithm, "algorithm");
70        this.slicer = checkNotNull(slicer, "slicer");
71     }
72  
73     @Override
74     public String execute(String container, Blob blob) {
75        String key = blob.getMetadata().getName();
76        ContentMetadata metadata = blob.getMetadata().getContentMetadata();
77        Payload payload = blob.getPayload();
78        Long length = payload.getContentMetadata().getContentLength();
79        checkNotNull(length,
80              "please invoke payload.getContentMetadata().setContentLength(length) prior to multipart upload");
81        long chunkSize = algorithm.calculateChunkSize(length);
82        int partCount = algorithm.getParts();
83        if (partCount > 0) {
84           ObjectMetadataBuilder builder = ObjectMetadataBuilder.create().key(key)
85              .contentType(metadata.getContentType())
86              .contentDisposition(metadata.getContentDisposition());
87           String uploadId = client.initiateMultipartUpload(container, builder.build());
88           try {
89              SortedMap<Integer, String> etags = Maps.newTreeMap();
90              int part;
91              while ((part = algorithm.getNextPart()) <= partCount) {
92                 prepareUploadPart(container, key, uploadId, part, payload, algorithm.getNextChunkOffset(), chunkSize,
93                       etags);
94              }
95              long remaining = algorithm.getRemaining();
96              if (remaining > 0) {
97                 prepareUploadPart(container, key, uploadId, part, payload, algorithm.getNextChunkOffset(), remaining,
98                       etags);
99              }
100             return client.completeMultipartUpload(container, key, uploadId, etags);
101          } catch (RuntimeException ex) {
102             client.abortMultipartUpload(container, key, uploadId);
103             throw ex;
104          }
105       } else {
106          // TODO: find a way to disable multipart. if we pass the original
107          // options, it goes into a stack overflow
108          return client.putObject(container, blobToObject.apply(blob));
109       }
110    }
111 
112    private void prepareUploadPart(String container, String key, String uploadId, int part, Payload payload,
113          long offset, long size, SortedMap<Integer, String> etags) {
114       Payload chunkedPart = slicer.slice(payload, offset, size);
115       String eTag = null;
116       try {
117          eTag = client.uploadPart(container, key, part, uploadId, chunkedPart);
118          etags.put(Integer.valueOf(part), eTag);
119       } catch (KeyNotFoundException e) {
120          // note that because of eventual consistency, the upload id may not be
121          // present yet we may wish to add this condition to the retry handler
122 
123          // we may also choose to implement ListParts and wait for the uploadId
124          // to become available there.
125          eTag = client.uploadPart(container, key, part, uploadId, chunkedPart);
126          etags.put(Integer.valueOf(part), eTag);
127       }
128    }
