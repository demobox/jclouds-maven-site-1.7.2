222    protected Optional<RoundRobinPool> getPoolById(final String poolId) {
223       return api(zoneName).list().firstMatch(new Predicate<RoundRobinPool>() {
224          public boolean apply(RoundRobinPool in) {
225             return in.getId().equals(poolId);
226          }
227       });
228    }
