42  /**
43   * @see <a href="https://ultra-api.ultradns.com:8443/UltraDNS_WS/v01?wsdl" />
44   * @see <a href="https://www.ultradns.net/api/NUS_API_XML_SOAP.pdf" />
45   * @author Adrian Cole
46   */
47  @RequestFilters(SOAPWrapWithPasswordAuth.class)
48  @VirtualHost
49  public interface ZoneApi {
50  
51     /**
52      * creates a primary zone and its supporting records (SOA, NS and A). The
53      * user who issues this request becomes the owner of this zone.
54      * 
55      * @param name
56      *           the fully qualified name of the new zone.
57      * @param accountId
58      *           the account to create the zone in
59      */
60     @Named("createPrimaryZone")
61     @POST
62     @Payload("<v01:createPrimaryZone><transactionID /><accountId>{accountId}</accountId><zoneName>{zoneName}</zoneName><forceImport>false</forceImport></v01:createPrimaryZone>")
63     void createInAccount(@PayloadParam("zoneName") String name, @PayloadParam("accountId") String accountId)
64           throws ResourceAlreadyExistsException;
65  
66     /**
67      * @param name
68      *           the fully-qualified name, including the trailing dot, of the
69      *           zone to get information about.
70      * @return null if not found
71      */
72     @Named("getGeneralPropertiesForZone")
73     @POST
74     @XMLResponseParser(ZonePropertiesHandler.class)
75     @Payload("<v01:getGeneralPropertiesForZone><zoneName>{zoneName}</zoneName></v01:getGeneralPropertiesForZone>")
76     @Fallback(NullOnNotFoundOr404.class)
77     @Nullable
78     ZoneProperties get(@PayloadParam("zoneName") String name);
79  
80     /**
81      * Lists all zones in the specified account.
82      * 
83      * @returns empty if no zones, or account doesn't exist
84      */
85     @Named("getZonesOfAccount")
86     @POST
87     @XMLResponseParser(ZoneListHandler.class)
88     @Payload("<v01:getZonesOfAccount><accountId>{accountId}</accountId><zoneType>all</zoneType></v01:getZonesOfAccount>")
89     FluentIterable<Zone> listByAccount(@PayloadParam("accountId") String accountId);
90  
91     /**
92      * Lists all zones in the specified account of type
93      * 
94      * @throws ResourceNotFoundException
95      *            if the account doesn't exist
96      */
97     @Named("getZonesOfAccount")
98     @POST
99     @XMLResponseParser(ZoneListHandler.class)
100    @Payload("<v01:getZonesOfAccount><accountId>{accountId}</accountId><zoneType>{zoneType}</zoneType></v01:getZonesOfAccount>")
101    FluentIterable<Zone> listByAccountAndType(@PayloadParam("accountId") String accountId,
102          @PayloadParam("zoneType") Type type) throws ResourceNotFoundException;
103 
104    /**
105     * deletes a zone and all its resource records
106     * 
107     * @param name
108     *           the fully-qualified name, including the trailing dot, of the
109     *           zone you want to delete.
110     * @return null if not found
111     */
112    @Named("deleteZone")
113    @POST
114    @Payload("<v01:deleteZone><transactionID /><zoneName>{zoneName}</zoneName></v01:deleteZone>")
115    @Fallback(VoidOnNotFoundOr404.class)
116    void delete(@PayloadParam("zoneName") String name);
117 }
