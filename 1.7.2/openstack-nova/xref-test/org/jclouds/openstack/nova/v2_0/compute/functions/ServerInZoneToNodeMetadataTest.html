207    private void checkHardwareAndImageStatus(Hardware expectedHardware, Hardware existingHardware,
208          String expectedImageId, OperatingSystem expectedOs, Image existingImage) {
209 
210       Set<Image> images = existingImage == null ? ImmutableSet.<Image> of() : ImmutableSet.of(existingImage);
211       Set<Hardware> hardwares = existingHardware == null ? ImmutableSet.<Hardware> of() : ImmutableSet
212             .of(existingHardware);
213       Server serverToConvert = new ParseServerTest().expected();
214 
215       ServerInZone serverInZoneToConvert = new ServerInZone(serverToConvert, "az-1.region-a.geo-1");
216 
217       ServerInZoneToNodeMetadata converter = new ServerInZoneToNodeMetadata(
218                NovaComputeServiceContextModule.toPortableNodeStatus, locationIndex, Suppliers
219                         .<Set<? extends Image>> ofInstance(images), Suppliers
220                         .<Set<? extends Hardware>> ofInstance(hardwares), namingConvention);
221 
222       NodeMetadata convertedNodeMetadata = converter.apply(serverInZoneToConvert);
223 
224       assertEquals(serverInZoneToConvert.slashEncode(), convertedNodeMetadata.getId());
225       assertEquals(serverToConvert.getId(), convertedNodeMetadata.getProviderId());
226 
227       assertEquals(convertedNodeMetadata.getLocation().getScope(), LocationScope.HOST);
228       assertEquals(convertedNodeMetadata.getLocation().getId(), "e4d909c290d0fb1ca068ffaddf22cbd0");
229 
230       assertEquals(convertedNodeMetadata.getLocation().getParent(), locationIndex.get().get("az-1.region-a.geo-1"));
231 
232       assertEquals(serverToConvert.getName(), convertedNodeMetadata.getName());
233       assertEquals(convertedNodeMetadata.getGroup(), "sample");
234 
235       assertEquals(convertedNodeMetadata.getImageId(), expectedImageId);
236       assertEquals(convertedNodeMetadata.getOperatingSystem(), expectedOs);
237 
238       assertEquals(convertedNodeMetadata.getHardware(), expectedHardware);
239 
240       assertEquals(NovaComputeServiceContextModule.toPortableNodeStatus.get(serverToConvert.getStatus()),
241                convertedNodeMetadata.getStatus());
242 
243       assertNotNull(convertedNodeMetadata.getPrivateAddresses());
244       assertEquals(convertedNodeMetadata.getPrivateAddresses(), ImmutableSet.of("10.176.42.16"));
245 
246       assertNotNull(convertedNodeMetadata.getPublicAddresses());
247       // note jclouds doesn't yet support ipv6 b/c not tested yet
248       assertEquals(convertedNodeMetadata.getPublicAddresses(), ImmutableSet.of("67.23.10.132", "67.23.10.131"));
249 
250       assertNotNull(convertedNodeMetadata.getUserMetadata());
251       assertEquals(convertedNodeMetadata.getUserMetadata(),
252             ImmutableMap.<String, String> of("Server Label", "Web Head 1", "Image Version", "2.1"));
253       
254       URI expectedURI = URI.create("http://servers.api.openstack.org/v1.1/1234/servers/52415800-8b69-11e0-9b19-734f6f006e54");
255       assertEquals(convertedNodeMetadata.getUri(), expectedURI);
256    }
