58  /**
59   * Nova implementation of {@link ImageExtension}
60   * 
61   * @author David Alves
62   *
63   */
64  @Singleton
65  public class NovaImageExtension implements ImageExtension {
66  
67     @Resource
68     @Named(ComputeServiceConstants.COMPUTE_LOGGER)
69     protected Logger logger = Logger.NULL;
70  
71     private final NovaApi novaApi;
72     private final ListeningExecutorService userExecutor;
73     private final Supplier<Set<? extends Location>> locations;
74     private final Predicate<AtomicReference<Image>> imageAvailablePredicate;
75  
76     @Inject
77     public NovaImageExtension(NovaApi novaApi, @Named(Constants.PROPERTY_USER_THREADS) ListeningExecutorService userExecutor,
78           @Memoized Supplier<Set<? extends Location>> locations,
79           @Named(TIMEOUT_IMAGE_AVAILABLE) Predicate<AtomicReference<Image>> imageAvailablePredicate) {
80        this.novaApi = checkNotNull(novaApi, "novaApi");
81        this.userExecutor = checkNotNull(userExecutor, "userExecutor");
82        this.locations = checkNotNull(locations, "locations");
83        this.imageAvailablePredicate = checkNotNull(imageAvailablePredicate, "imageAvailablePredicate");
84     }
85  
86     @Override
87     public ImageTemplate buildImageTemplateFromNode(String name, final String id) {
88        ZoneAndId zoneAndId = ZoneAndId.fromSlashEncoded(id);
89        Server server = novaApi.getServerApiForZone(zoneAndId.getZone()).get(zoneAndId.getId());
90        if (server == null)
91           throw new NoSuchElementException("Cannot find server with id: " + zoneAndId);
92        CloneImageTemplate template = new ImageTemplateBuilder.CloneImageTemplateBuilder().nodeId(id).name(name).build();
93        return template;
94     }
95  
96     @Override
97     public ListenableFuture<Image> createImage(ImageTemplate template) {
98        checkState(template instanceof CloneImageTemplate,
99                 " openstack-nova only supports creating images through cloning.");
100       CloneImageTemplate cloneTemplate = (CloneImageTemplate) template;
101       ZoneAndId sourceImageZoneAndId = ZoneAndId.fromSlashEncoded(cloneTemplate.getSourceNodeId());
102 
103       String newImageId = novaApi.getServerApiForZone(sourceImageZoneAndId.getZone()).createImageFromServer(
104                cloneTemplate.getName(), sourceImageZoneAndId.getId());
105 
106       final ZoneAndId targetImageZoneAndId = ZoneAndId.fromZoneAndId(sourceImageZoneAndId.getZone(), newImageId);
107 
108       logger.info(">> Registered new Image %s, waiting for it to become available.", newImageId);
109       
110       final AtomicReference<Image> image = Atomics.newReference(new ImageBuilder()
111             .location(find(locations.get(), idEquals(targetImageZoneAndId.getZone())))
112             .id(targetImageZoneAndId.slashEncode())
113             .providerId(targetImageZoneAndId.getId())
114             .description(cloneTemplate.getName())
115             .operatingSystem(OperatingSystem.builder().description(cloneTemplate.getName()).build())
116             .status(Image.Status.PENDING).build());
117 
118       return userExecutor.submit(new Callable<Image>() {
119          @Override
120          public Image call() throws Exception {
121             if (imageAvailablePredicate.apply(image))
122                return image.get();
123             // TODO: get rid of the expectation that the image will be available, as it is very brittle
124             throw new UncheckedTimeoutException("Image was not created within the time limit: " + image.get());
125          }
126       });
127    }
128 
129    @Override
130    public boolean deleteImage(String id) {
131       ZoneAndId zoneAndId = ZoneAndId.fromSlashEncoded(id);
132       try {
133          this.novaApi.getImageApiForZone(zoneAndId.getZone()).delete(zoneAndId.getId());
134       } catch (Exception e) {
135          return false;
136       }
137       return true;
138    }
