111 	private void decryptAndVerify(Crypto crypto)
112 			throws InvalidKeySpecException, NoSuchAlgorithmException,
113 			NoSuchPaddingException, InvalidKeyException,
114 			IllegalBlockSizeException, BadPaddingException {
115 		
116 		KeySpec keySpec = Pems.privateKeySpec(PRIVATE_KEY);
117 		KeyFactory kf = crypto.rsaKeyFactory();
118 		PrivateKey privKey = kf.generatePrivate(keySpec);
119 
120 		Cipher cipher = crypto.cipher("RSA");
121 		cipher.init(Cipher.DECRYPT_MODE, privKey);
122 		byte[] cipherText = base64().decode(ENCRYPTED_PASSWORD);
123 		byte[] plainText = cipher.doFinal(cipherText);
124 		String password = new String(plainText, Charsets.US_ASCII);
