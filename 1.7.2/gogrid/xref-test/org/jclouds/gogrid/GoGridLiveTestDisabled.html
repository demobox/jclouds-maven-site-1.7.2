181    public void testJobs() {
182       final String nameOfServer = "Server" + String.valueOf(new Date().getTime()).substring(6);
183       serversToDeleteAfterTheTests.add(nameOfServer);
184 
185       Set<Ip> availableIps = api.getIpServices().getUnassignedPublicIpList();
186 
187       String ram = Iterables.get(api.getServerServices().getRamSizes(), 0).getName();
188 
189       Server createdServer = api.getServerServices().addServer(nameOfServer,
190                "GSI-f8979644-e646-4711-ad58-d98a5fa3612c", ram, Iterables.getLast(availableIps).getIp());
191 
192       assert serverLatestJobCompleted.apply(createdServer);
193 
194       // restart the server
195       api.getServerServices().power(nameOfServer, PowerCommand.RESTART);
196 
197       Set<Job> jobs = api.getJobServices().getJobsForObjectName(nameOfServer);
198 
199       Job latestJob = Iterables.getLast(jobs);
200       Long latestJobId = latestJob.getId();
201 
202       Job latestJobFetched = Iterables.getOnlyElement(api.getJobServices().getJobsById(latestJobId));
203 
204       assert latestJob.equals(latestJobFetched) : "Job and its representation found by ID don't match";
205 
206       long[] idsOfAllJobs = new long[jobs.size()];
207       int i = 0;
208       for (Job job : jobs) {
209          idsOfAllJobs[i++] = job.getId();
210       }
211 
212       Set<Job> jobsFetched = api.getJobServices().getJobsById(idsOfAllJobs);
213       assert jobsFetched.size() == jobs.size() : format(
214                "Number of jobs fetched by ids doesn't match the number of jobs "
215                         + "requested. Requested/expected: %d. Found: %d.", jobs.size(), jobsFetched.size());
216 
217       // delete the server
218       api.getServerServices().deleteByName(nameOfServer);
219    }

224    @Test(enabled = true)
225    public void testLoadBalancerLifecycle() {
226       int lbCountBeforeTest = api.getLoadBalancerServices().getLoadBalancerList().size();
227 
228       final String nameOfLoadBalancer = "LoadBalancer" + String.valueOf(new Date().getTime()).substring(6);
229       loadBalancersToDeleteAfterTest.add(nameOfLoadBalancer);
230 
231       Set<Ip> availableIps = api.getIpServices().getUnassignedPublicIpList();
232 
233       if (availableIps.size() < 4)
234          throw new SkipException("Not enough available IPs (4 needed) to run the test");
235       Iterator<Ip> ipIterator = availableIps.iterator();
236       Ip vip = ipIterator.next();
237       Ip realIp1 = ipIterator.next();
238       Ip realIp2 = ipIterator.next();
239       Ip realIp3 = ipIterator.next();
240 
241       AddLoadBalancerOptions options = new AddLoadBalancerOptions.Builder().create(LoadBalancerType.LEAST_CONNECTED,
242                LoadBalancerPersistenceType.SOURCE_ADDRESS);
243       LoadBalancer createdLoadBalancer = api.getLoadBalancerServices().addLoadBalancer(nameOfLoadBalancer,
244                IpPortPair.builder().ip(vip).port(80).build(), Arrays.asList(IpPortPair.builder().ip(realIp1).port(80).build(),
245                IpPortPair.builder().ip(realIp2).port(80).build()),
246                options);
247       assertNotNull(createdLoadBalancer);
248       assert loadBalancerLatestJobCompleted.apply(createdLoadBalancer);
249 
250       // get load balancer by name
251       Set<LoadBalancer> response = api.getLoadBalancerServices().getLoadBalancersByName(nameOfLoadBalancer);
252       assert response.size() == 1;
253       createdLoadBalancer = Iterables.getOnlyElement(response);
254       assertNotNull(createdLoadBalancer.getRealIpList());
255       assertEquals(createdLoadBalancer.getRealIpList().size(), 2);
256       assertNotNull(createdLoadBalancer.getVirtualIp());
257       assertEquals(createdLoadBalancer.getVirtualIp().getIp().getIp(), vip.getIp());
258 
259       LoadBalancer editedLoadBalancer = api.getLoadBalancerServices().editLoadBalancerNamed(nameOfLoadBalancer,
260                Arrays.asList(IpPortPair.builder().ip(realIp3).port(8181).build()));
261       assert loadBalancerLatestJobCompleted.apply(editedLoadBalancer);
262       assertNotNull(editedLoadBalancer.getRealIpList());
263       assertEquals(editedLoadBalancer.getRealIpList().size(), 1);
264       assertEquals(Iterables.getOnlyElement(editedLoadBalancer.getRealIpList()).getIp().getIp(), realIp3.getIp());
265 
266       int lbCountAfterAddingOneServer = api.getLoadBalancerServices().getLoadBalancerList().size();
267       assert lbCountAfterAddingOneServer == lbCountBeforeTest + 1 : "There should be +1 increase in the number of load balancers since the test started";
268 
269       // delete the load balancer
270       api.getLoadBalancerServices().deleteByName(nameOfLoadBalancer);
271 
272       Set<Job> jobs = api.getJobServices().getJobsForObjectName(nameOfLoadBalancer);
273       assert "DeleteLoadBalancer".equals(Iterables.getLast(jobs).getCommand().getName());
274 
275       assert loadBalancerLatestJobCompleted.apply(createdLoadBalancer);
276 
277       int lbCountAfterDeletingTheServer = api.getLoadBalancerServices().getLoadBalancerList().size();
278       assert lbCountAfterDeletingTheServer == lbCountBeforeTest : "There should be the same # of load balancers as since the test started";
279    }

341       assertEquals(api.getServerServices().getServerCredentials(createdServer.getId()), instanceCredentials);
