17  package org.jclouds.cloudstack.features;
18  
19  import javax.inject.Named;
20  import javax.ws.rs.Consumes;
21  import javax.ws.rs.GET;
22  import javax.ws.rs.QueryParam;
23  import javax.ws.rs.core.MediaType;
24  
25  import org.jclouds.Fallbacks.NullOnNotFoundOr404;
26  import org.jclouds.cloudstack.domain.Account;
27  import org.jclouds.cloudstack.filters.AuthenticationFilter;
28  import org.jclouds.cloudstack.options.CreateAccountOptions;
29  import org.jclouds.cloudstack.options.UpdateAccountOptions;
30  import org.jclouds.rest.annotations.Fallback;
31  import org.jclouds.rest.annotations.QueryParams;
32  import org.jclouds.rest.annotations.RequestFilters;
33  import org.jclouds.rest.annotations.SelectJson;
34  
35  /**
36   * Provides synchronous access to CloudStack Account features available to Global
37   * Admin users.
38   *
39   * @author Adrian Cole, Andrei Savu
40   * @see <a href=
41   *      "http://download.cloud.com/releases/2.2.0/api_2.2.12/TOC_Global_Admin.html"
42   *      />
43   */
44  @RequestFilters(AuthenticationFilter.class)
45  @QueryParams(keys = "response", values = "json")
46  public interface GlobalAccountApi extends DomainAccountApi {
47  
48     /**
49      * Create a new Cloudstack account
50      *
51      * @param userName unique username.
52      * @param accountType type of account
53      * @param email
54      * @param firstName
55      * @param lastName
56      * @param hashedPassword
57      *          Hashed password (Default is MD5). If you wish to use any other
58      *          hashing algorithm, you would need to write a custom authentication adapter See Docs section.
59      * @param options
60      *          optional parameters
61      * @return
62      */
63     @Named("createAccount")
64     @GET
65     @QueryParams(keys = "command", values = "createAccount")
66     @SelectJson("account")
67     @Consumes(MediaType.APPLICATION_JSON)
68     @Fallback(NullOnNotFoundOr404.class)
69     Account createAccount(@QueryParam("username") String userName,
70        @QueryParam("accounttype") Account.Type accountType, @QueryParam("email") String email,
71        @QueryParam("firstname") String firstName, @QueryParam("lastname") String lastName,
72        @QueryParam("password") String hashedPassword, CreateAccountOptions... options);
73  
74     /**
75      * Update an existing account
76      *
77      * @param accountName the current account name
78      * @param domainId the ID of the domain were the account exists
79      * @param newName new name for the account
80      * @param options optional arguments
81      * @return
82      */
83     @Named("updateAccount")
84     @GET
85     @QueryParams(keys = "command", values = "updateAccount")
86     @SelectJson("account")
87     @Consumes(MediaType.APPLICATION_JSON)
88     @Fallback(NullOnNotFoundOr404.class)
89     Account updateAccount(@QueryParam("account") String accountName,
90        @QueryParam("domainid") String domainId, @QueryParam("newname") String newName, UpdateAccountOptions... options);
91     
92     /**
93      * Delete an account with the specified ID
94      *
95      * @param accountId
96      * @return
97      */
98     @Named("deleteAccount")
99     @GET
100    @QueryParams(keys = "command", values = "deleteAccount")
101    @Consumes(MediaType.APPLICATION_JSON)
102    @Fallback(NullOnNotFoundOr404.class)
103    void deleteAccount(@QueryParam("id") String id);
104 }
