330   public void destroyNode(String id) {
331      String virtualMachineId = id;
332      // There was a bug in 2.2.12 release happening when static nat IP address
333      // was being released, and corresponding firewall rules were left behind.
334      // So next time the same IP is allocated, it might be not be static nat
335      // enabled, but there are still rules associated with it. And when you try
336      // to release this IP, the release will fail.
337      //
338      // The bug was fixed in 2.2.13 release only, and the current system wasn't
339      // updated yet.
340      //
341      // To avoid the issue, every time you release a static nat ip address, do
342      // the following:
343 
344      // 1) Delete IP forwarding rules associated with IP.
345      Set<String> ipAddresses = deleteIPForwardingRulesForVMAndReturnDistinctIPs(virtualMachineId);
346 
347      // 2) Delete firewall rules associated with IP.
348      ipAddresses.addAll(deleteFirewallRulesForVMAndReturnDistinctIPs(virtualMachineId));
349 
350      // 3) Disable static nat rule for the IP.
351      disableStaticNATOnIPAddresses(ipAddresses);
352 
353      // 4) Only after 1 and 2 release the IP address.
354      disassociateIPAddresses(ipAddresses);
355 
356      destroyVirtualMachine(virtualMachineId);
357 
358      vmToRules.invalidate(virtualMachineId);
359   }
