63     public void testCreateLoadBalancer() throws Exception {
64        for (String zone: api.getConfiguredZones()) {
65           Logger.getAnonymousLogger().info("starting lb in region " + zone);
66           
67           LoadBalancer lb = api.getLoadBalancerApiForZone(zone).create(
68                 CreateLoadBalancer.builder()
69                       .name(prefix + "-" + zone)
70                       .protocol("HTTP")
71                       .port(80)
72                       .virtualIPType(Type.PUBLIC)
73                       .node(AddNode.builder()
74                             .address("192.168.1.1")
75                             .port(8080)
76                             .build())
77                       .build());
78           checkLBInRegion(zone, lb, prefix + "-" + zone);
79           
80           assertEquals(lb.getStatus(), LoadBalancer.Status.BUILD);
81           
82           lbs.add(lb);
83           
84           assertTrue(awaitAvailable(api.getLoadBalancerApiForZone(lb.getRegion())).apply(lb));
85  
86           LoadBalancer newLb = api.getLoadBalancerApiForZone(zone).get(lb.getId());
87           checkLBInRegion(zone, newLb, prefix + "-" + zone);
88           
89           assertEquals(newLb.getStatus(), LoadBalancer.Status.ACTIVE);
90        }
91     }

108    @Test(dependsOnMethods = "testUpdateLoadBalancer")
109    public void testListLoadBalancers() throws Exception {
110       for (String zone: api.getConfiguredZones()) {
111 
112          Set<LoadBalancer> response = api.getLoadBalancerApiForZone(zone).list().concat().toSet();
113          
114          assertNotNull(response);
115          assertTrue(response.size() >= 0);
116          
117          for (LoadBalancer lb: response) {
118             if (!lbs.contains(lb))
119                continue;
120             assert lb.getRegion() != null : lb;
121             assert lb.getName() != null : lb;
122             assert lb.getId() != -1 : lb;
123             assert lb.getProtocol() != null : lb;
124             assert lb.getPort() != -1 : lb;
125             assert lb.getStatus() != null : lb;
126             assert lb.getCreated() != null : lb;
127             assert lb.getUpdated() != null : lb;
128             assert lb.getVirtualIPs().size() > 0 : lb;
129             // node info not available during list;
130             assert lb.getNodes().size() == 0 : lb;
131 
132             LoadBalancer getDetails = api.getLoadBalancerApiForZone(zone).get(lb.getId());
133             
134             try {
135                assertEquals(getDetails.getRegion(), lb.getRegion());
136                assertEquals(getDetails.getName(), lb.getName());
137                assertEquals(getDetails.getId(), lb.getId());
138                assertEquals(getDetails.getProtocol(), lb.getProtocol());
139                assertEquals(getDetails.getPort(), lb.getPort());
140                assertEquals(getDetails.getStatus(), lb.getStatus());
141                assertEquals(getDetails.getCreated(), lb.getCreated());
142                assertEquals(getDetails.getUpdated(), lb.getUpdated());
143                assertEquals(getDetails.getVirtualIPs(), lb.getVirtualIPs());
144                // node info not available during list;
145                assert getDetails.getNodes().size() > 0 : lb;
146             } catch (AssertionError e) {
147                throw new AssertionError(String.format("%s\n%s - %s", e.getMessage(), getDetails, lb));
148             }
149          }
150       }
151    }

153    @Test(dependsOnMethods = "testListLoadBalancers")
154    public void testLoadBalancerMetadata() throws Exception {
155       for (LoadBalancer lb: lbs) {
156          Map<String, String> metadataMap = ImmutableMap.<String, String> of(
157                "key1", "value1",
158                "key2", "value2",
159                "key3", "value3");
160          
161          Metadata metadata = api.getLoadBalancerApiForZone(lb.getRegion()).createMetadata(lb.getId(), metadataMap);
162          assertEquals(metadata, getExpectedMetadata());
163          assertTrue(awaitAvailable(api.getLoadBalancerApiForZone(lb.getRegion())).apply(lb));
164 
165          metadata = api.getLoadBalancerApiForZone(lb.getRegion()).getMetadata(lb.getId());
166          assertEquals(metadata, getExpectedMetadata());
167 
168          assertTrue(api.getLoadBalancerApiForZone(lb.getRegion()).updateMetadatum(lb.getId(), metadata.getId("key1"), "key1-updated"));
169          assertTrue(awaitAvailable(api.getLoadBalancerApiForZone(lb.getRegion())).apply(lb));
170          metadata = api.getLoadBalancerApiForZone(lb.getRegion()).getMetadata(lb.getId());
171          assertEquals(metadata.get("key1"), "key1-updated");
172 
173          assertTrue(api.getLoadBalancerApiForZone(lb.getRegion()).deleteMetadatum(lb.getId(), metadata.getId("key1")));
174          assertTrue(awaitAvailable(api.getLoadBalancerApiForZone(lb.getRegion())).apply(lb));
175          metadata = api.getLoadBalancerApiForZone(lb.getRegion()).getMetadata(lb.getId());
176          assertNull(metadata.get("key1"));
177 
178          assertTrue(api.getLoadBalancerApiForZone(lb.getRegion()).deleteMetadata(lb.getId(), 
179                ImmutableList.<Integer> of(metadata.getId("key2"), metadata.getId("key3"))));
180          assertTrue(awaitAvailable(api.getLoadBalancerApiForZone(lb.getRegion())).apply(lb));
181          metadata = api.getLoadBalancerApiForZone(lb.getRegion()).getMetadata(lb.getId());
182          assertEquals(metadata.size(), 0);
183       }
184    }
