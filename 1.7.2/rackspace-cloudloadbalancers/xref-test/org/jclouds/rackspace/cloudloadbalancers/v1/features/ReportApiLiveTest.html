47  @Test(groups = "live", singleThreaded = true, testName = "ReportApiLiveTest")
48  public class ReportApiLiveTest extends BaseCloudLoadBalancersApiLiveTest {
49     private LoadBalancer lb;
50     private String zone;
51  
52     public void testCreateLoadBalancer() {
53        AddNode addNode = AddNode.builder().address("192.168.1.1").port(8080).build();
54        CreateLoadBalancer createLB = CreateLoadBalancer.builder()
55              .name(prefix+"-jclouds").protocol("HTTP").port(80).virtualIPType(Type.PUBLIC).node(addNode).build(); 
56  
57        zone = Iterables.getFirst(api.getConfiguredZones(), null);
58        lb = api.getLoadBalancerApiForZone(zone).create(createLB);
59        
60        assertTrue(awaitAvailable(api.getLoadBalancerApiForZone(zone)).apply(lb));
61     }
62  
63     @Test(dependsOnMethods = "testCreateLoadBalancer")
64     public void testReports() throws Exception {
65        Calendar calendar = Calendar.getInstance();
66        calendar.add(Calendar.DATE, -1);      
67        Date yesterday = calendar.getTime();
68        Date today = new Date();
69  
70        FluentIterable<LoadBalancer> loadBalancers = api.getReportApiForZone(zone).listBillableLoadBalancers(yesterday, today).concat();
71        assertNotNull(loadBalancers);
72        
73        HistoricalUsage historicalUsage = api.getReportApiForZone(zone).getHistoricalUsage(yesterday, today);
74        assertNotEquals(historicalUsage.getAccountId(), 0);
75  
76        FluentIterable<LoadBalancerUsage> loadBalancerUsages = api.getReportApiForZone(zone).listLoadBalancerUsage(lb.getId(), yesterday, today).concat();
77        assertNotNull(loadBalancerUsages);
78  
79        loadBalancerUsages = api.getReportApiForZone(zone).listCurrentLoadBalancerUsage(lb.getId()).concat();
80        assertNotNull(loadBalancerUsages);
81        
82        try {
83           LoadBalancerStats loadBalancerStats = api.getReportApiForZone(zone).getLoadBalancerStats(lb.getId());
84           assertNotNull(loadBalancerStats);
85        }
86        catch (HttpResponseException e) {
87           // CLB sometimes doesn't like it when you get stats on a newly created LB so ignore a 500
88           if (e.getResponse().getStatusCode() != 500) {
89              throw e;
90           }
91        }
92        
93        Iterable<Protocol> protocols = api.getReportApiForZone(zone).listProtocols();
94        assertTrue(!Iterables.isEmpty(protocols));
95  
96        Iterable<String> algorithms = api.getReportApiForZone(zone).listAlgorithms();
97        assertTrue(!Iterables.isEmpty(algorithms));
98     }
99  
100    @Override
101    @AfterGroups(groups = "live")
102    protected void tearDown() {
103       assertTrue(awaitAvailable(api.getLoadBalancerApiForZone(zone)).apply(lb));
104       api.getLoadBalancerApiForZone(zone).delete(lb.getId());
105       assertTrue(awaitDeleted(api.getLoadBalancerApiForZone(zone)).apply(lb));
106       super.tearDown();
107    }
