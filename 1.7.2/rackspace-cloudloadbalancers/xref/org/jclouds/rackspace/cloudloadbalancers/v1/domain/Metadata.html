17  package org.jclouds.rackspace.cloudloadbalancers.v1.domain;
18  
19  import java.util.Map;
20  
21  import com.google.common.collect.ForwardingMap;
22  import static com.google.common.collect.Maps.newLinkedHashMap;
23  import static com.google.common.collect.Sets.newHashSet;
24  
25  /**
26   * Key and value must be 256 characters or less. All UTF-8 characters are valid.
27   * </p>
28   * Use the *Id methods when you need to get a metadata id for updating and removal.
29   * 
30   * @author Everett Toews
31   */
32  public class Metadata extends ForwardingMap<String, String> {
33     private final Map<String, String> metadata = newLinkedHashMap(); // Implementing class must be sorted.
34     private final Map<String, Integer> keyToId = newLinkedHashMap();
35     
36     public Metadata(Metadata metadata) {
37        super();
38        this.metadata.putAll(metadata);
39     }
40     
41     public Metadata() {
42        super();
43     }
44  
45     @Override
46     protected Map<String, String> delegate() {
47        return metadata;
48     }
49     
50     public int getId(String key) {
51        return keyToId.get(key);
52     }
53     
54     public Integer putId(String key, int id) {
55        return keyToId.put(key, id);
56     }
57  
58     public Iterable<Integer> getIds() {
59        return newHashSet(keyToId.values());
60     }
61  }
