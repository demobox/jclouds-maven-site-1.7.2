21  import javax.annotation.Resource;
22  import javax.inject.Inject;

24  import org.jclouds.logging.Logger;
25  import org.jclouds.rackspace.cloudloadbalancers.v1.domain.AccessRuleWithId;

27  import org.jclouds.rackspace.cloudloadbalancers.v1.domain.LoadBalancer.Builder;
28  import org.jclouds.rackspace.cloudloadbalancers.v1.domain.Metadata;

31  import com.google.common.base.Function;
32  import com.google.common.collect.ImmutableSet;
33  import com.google.common.collect.Iterables;
34  import com.google.inject.assistedinject.Assisted;

36  /**
37   * @author Adrian Cole
38   */
39  public class ConvertLB implements Function<LB, LoadBalancer> {
40     private static final String LOAD_BALANCERS = "loadbalancers";
41  
42     public interface Factory {
43        ConvertLB createForEndpointAndRegion(URI endpoint, String region);
44     }
45  
46     @Resource
47     protected Logger logger = Logger.NULL;
48  
49     private final String region;
50     private final URI endpoint;
51  
52     @Inject
53     ConvertLB(@Assisted URI endpoint, @Assisted String region) {
54        this.endpoint = endpoint;
55        this.region = region.toUpperCase();
56     }
57  
58     @Override
59     public LoadBalancer apply(LB lb) {
60        try {
61           Builder builder = LoadBalancer.builder().id(lb.id).region(region).status(lb.status).name(lb.getName())
62                 .protocol(lb.getProtocol()).port(lb.getPort()).nodeCount(lb.nodeCount).nodes(lb.getNodes())
63                 .timeout(lb.getTimeout()).algorithm(lb.getAlgorithm()).halfClosed(lb.isHalfClosed())
64                 .sessionPersistenceType(lb.getSessionPersistenceType()).connectionLogging(lb.isConnectionLogging())
65                 .connectionThrottle(lb.getConnectionThrottle()).healthMonitor(lb.getHealthMonitor());
66  
67           if (lb.cluster.size() == 1)
68              builder.clusterName(Iterables.get(lb.cluster.values(), 0));
69           if (lb.created.size() == 1)
70              builder.created(Iterables.get(lb.created.values(), 0));
71           if (lb.updated.size() == 1)
72              builder.updated(Iterables.get(lb.updated.values(), 0));
73           if (lb.contentCaching.size() == 1)
74              builder.contentCaching(Iterables.get(lb.contentCaching.values(), 0));
75           if (lb.sslTermination != null)
76              builder.sslTermination(lb.sslTermination);
77           if (lb.sourceAddresses != null)
78              builder.sourceAddresses(lb.sourceAddresses);
79           if (lb.accessList == null)
80              builder.accessRules(ImmutableSet.<AccessRuleWithId> of());
81           else
82              builder.accessRules(lb.accessList);
83           if (lb.virtualIps == null)
84              builder.virtualIPs(ImmutableSet.<VirtualIPWithId> of());
85           else
86              builder.virtualIPs(lb.virtualIps);
87           if (lb.metadata == null)
88              builder.metadata(new Metadata());
89           else
90              builder.metadata(ParseMetadata.transformCLBMetadataToMetadata(lb.metadata));
91           
92           int indexOfLB = endpoint.toString().lastIndexOf(LOAD_BALANCERS); 
93           String path = endpoint.toString().substring(0, indexOfLB + LOAD_BALANCERS.length()); 
94           builder.uri(URI.create(path + "/" + lb.id));
95  
96           return builder.build();
97        }
98        catch (NullPointerException e) {
99           logger.warn(e, "nullpointer found parsing %s", lb);
100          throw e;
101       }
102    }
