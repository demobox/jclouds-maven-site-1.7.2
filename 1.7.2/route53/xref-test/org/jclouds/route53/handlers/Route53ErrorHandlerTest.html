45  @Test(groups = "unit", singleThreaded = true, testName = "Route53ErrorHandlerTest" )
46  public class Route53ErrorHandlerTest {
47     Route53ErrorHandler function = Guice.createInjector(new SaxParserModule()).getInstance(Route53ErrorHandler.class);
48  
49     HttpRequest request = HttpRequest.builder().method(POST)
50           .endpoint("https://route53.amazonaws.com/2012-02-29/hostedzone/Z1PA6795UKMFR9/rrset")
51           .addHeader(HOST, "route53.amazonaws.com")
52           .addHeader(DATE, "Mon, 21 Jan 02013 19:29:03 -0800")
53           .addHeader("X-Amzn-Authorization", "AWS3-HTTPS AWSAccessKeyId=identity,Algorithm=HmacSHA256,Signature=pylxNiLcrsjNRZOsxyT161JCwytVPHyc2rFfmNCuZKI=")
54           .payload(payloadFromResource("/batch_rrs_request.xml")).build();
55     HttpCommand command = new HttpCommand(request);
56  
57     @Test
58     public void testInvalidChangeBatchException() throws IOException {
59        HttpResponse response = HttpResponse.builder().statusCode(BAD_REQUEST.getStatusCode())
60                                                      .payload(payloadFromResource("/invalid_change_batch.xml")).build();
61        function.handleError(command, response);
62  
63        InvalidChangeBatchException exception = InvalidChangeBatchException.class.cast(command.getException());
64  
65        assertEquals(exception.getMessages(), ImmutableSet.of(
66              "Tried to create resource record set duplicate.example.com. type A, but it already exists",
67              "Tried to delete resource record set noexist.example.com. type A, but it was not found"));
68     }
69  
70     @Test
71     public void testDeleteNotFound() throws IOException {
72  
73        HttpResponse response = HttpResponse.builder().statusCode(BAD_REQUEST.getStatusCode())
74              .payload(
75                    payloadFromStringWithContentType(
76                          "<ErrorResponse><Error><Type>Sender</Type><Code>InvalidChangeBatch</Code>"
77                                + "<Message>Tried to delete resource record set krank.foo.bar., type TXT but it was not found</Message>"
78                                + "</Error></ErrorResponse>", "application/xml")).build();
79  
80        function.handleError(command, response);
81  
82        assertEquals(command.getException().getClass(), ResourceNotFoundException.class);
83        assertEquals(command.getException().getMessage(), "Tried to delete resource record set krank.foo.bar., type TXT but it was not found");
84  
85        AWSResponseException exception = AWSResponseException.class.cast(command.getException().getCause());
86  
87        assertEquals(exception.getError().getCode(), "InvalidChangeBatch");
88     }
89  
90     private Payload payloadFromResource(String resource) {
91        try {
92           return payloadFromStringWithContentType(toStringAndClose(getClass().getResourceAsStream(resource)),
93                 "application/xml");
94        } catch (IOException e) {
95           throw propagate(e);
96        }
97     }
98  }
