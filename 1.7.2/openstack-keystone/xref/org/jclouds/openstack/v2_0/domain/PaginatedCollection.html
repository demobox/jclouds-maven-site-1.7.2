17  package org.jclouds.openstack.v2_0.domain;
18  
19  import com.google.common.base.Optional;
20  import com.google.common.collect.ImmutableSet;
21  import com.google.common.collect.Multimap;
22  import org.jclouds.collect.IterableWithMarker;
23  import org.jclouds.javax.annotation.Nullable;
24  import org.jclouds.openstack.v2_0.options.PaginationOptions;
25  
26  import java.util.Iterator;
27  
28  import static org.jclouds.http.utils.Queries.queryParser;
29  
30  /**
31   * Base class for a paginated collection in OpenStack.
32   *
33   * @see <a
34   *      href="http://docs.openstack.org/api/openstack-identity-service/2.0/content/Paginated_Collections-d1e325.html">
35   *      docs</a>
36   * @author Everett Toews
37   */
38  public class PaginatedCollection<T> extends IterableWithMarker<T> {
39     private final Iterable<T> resources;
40     private final Iterable<Link> links;
41     private final Integer totalEntries;
42  
43     protected PaginatedCollection(@Nullable Iterable<T> resources, @Nullable Iterable<Link> links,
44                                   @Nullable Integer totalEntries) {
45        this.resources = resources != null ? resources : ImmutableSet.<T> of();
46        this.links = links != null ? links : ImmutableSet.<Link> of();
47        this.totalEntries = totalEntries;
48     }
49  
50     protected PaginatedCollection(Iterable<T> resources, Iterable<Link> links) {
51        this(resources, links, null);
52     }
53  
54     @Override
55     public Iterator<T> iterator() {
56        return resources.iterator();
57     }
58  
59     /**
60      * links that relate to this collection
61      */
62     public Iterable<Link> getLinks() {
63        return links;
64     }
65  
66     /**
67      * @return The total number of entries in this collection, if that information is present.
68      */
69     public Optional<Integer> getTotalEntries() {
70        return Optional.fromNullable(totalEntries);
71     }
72  
73     public PaginationOptions nextPaginationOptions() {
74        return PaginationOptions.class.cast(nextMarker().get());
75     }
76  
77     @Override
78     public Optional<Object> nextMarker() {
79        for (Link link: getLinks()) {
80           if (Link.Relation.NEXT == link.getRelation()) {
81              return Optional.of(toPaginationOptions(link));
82           }
83        }
84  
85        return Optional.absent();
86     }
87  
88     private Object toPaginationOptions(Link link) {
89        Multimap<String, String> queryParams = queryParser().apply(link.getHref().getRawQuery());
90        PaginationOptions paginationOptions = PaginationOptions.Builder.queryParameters(queryParams);
91  
92        return paginationOptions;
93     }
94  }
