294   protected void cleanUpIncidentalResources(final String region, final String group) {
295      // For issue #445, tries to delete security groups first: ec2 throws exception if in use, but
296      // deleting a key pair does not.
297      // This is "belt-and-braces" because deleteKeyPair also does extractIdsFromInstances & usingKeyPairAndNotDead
298      // for us to check if any instances are using the key-pair before we delete it. 
299      // There is (probably?) still a race if someone is creating instances at the same time as deleting them: 
300      // we may delete the key-pair just when the node-being-created was about to rely on the incidental 
301      // resources existing.
302 
303      // Also in #445, in aws-ec2 the deleteSecurityGroup sometimes fails after terminating the final VM using a 
304      // given security group, if called very soon after the VM's state reports terminated. Empirically, it seems that
305      // waiting a small time (e.g. enabling logging or debugging!) then the tests pass. We therefore retry.
306      // TODO: this could be moved to a config module, also the narrative above made more concise
307      retry(new Predicate<RegionAndName>() {
308         public boolean apply(RegionAndName input) {
309            try {
310               logger.debug(">> deleting incidentalResources(%s)", input);
311               deleteSecurityGroup(input.getRegion(), input.getName());
312               deleteKeyPair(input.getRegion(), input.getName()); // not executed if securityGroup was in use
313               logger.debug("<< deleted incidentalResources(%s)", input);
314               return true;
315            } catch (IllegalStateException e) {
316               logger.debug("<< inUse incidentalResources(%s)", input);
317               return false;
318            }
319         }
320      }, SECONDS.toMillis(3), 50, 1000, MILLISECONDS).apply(new RegionAndName(region, group));
321   }
