65        public JsonElement serialize(SessionCredentials src, Type typeOfSrc, JsonSerializationContext context) {

74     private static class GeoRegionGroupTypeAdapter implements JsonDeserializer<GeoRegionGroup> {
75  
76        @Override
77        public GeoRegionGroup deserialize(JsonElement json, Type typeOfT, JsonDeserializationContext context)
78              throws JsonParseException {
79           CreepyGeoRegionGroup creepyGeoRegionGroup = context.deserialize(json, CreepyGeoRegionGroup.class);
80           GeoRegionGroup.Builder builder = GeoRegionGroup.builder();
81           builder.name(creepyGeoRegionGroup.name);
82           builder.serviceName(creepyGeoRegionGroup.service_name);
83           builder.countries(creepyGeoRegionGroup.countries);
84           ImmutableList.Builder<RecordSet> rsets = ImmutableList.builder();
85           for (Entry<String, List<Map<String, Object>>> entry : creepyGeoRegionGroup.rdata.entrySet()) {
86              if (entry.getValue().isEmpty())
87                 continue;
88              // ex. spf_rdata -> SPF
89              String type = entry.getKey().substring(0, entry.getKey().indexOf('_')).toUpperCase();
90              // ex. dhcid_ttl
91              int ttl = creepyGeoRegionGroup.ttl.get(type.toLowerCase() + "_ttl");
92              RecordSet.Builder rset = RecordSet.builder();
93              rset.type(type);
94              rset.ttl(ttl);
95  
96              // weight is only present for a couple record types
97              List<Integer> weights = creepyGeoRegionGroup.weight.get(type.toLowerCase() + "_weight");
98              if (weights == null)
99                 weights = ImmutableList.of();
100 
101             List<String> labels = creepyGeoRegionGroup.label.get(type.toLowerCase() + "_label");
102             for (int i = 0; i < entry.getValue().size(); i++) {
103                Builder elementBuilder = Value.builder().rdata(entry.getValue().get(i));
104                // chance of index out of bounds
105                if (i < labels.size())
106                   elementBuilder.label(labels.get(i));
107                if (i < weights.size())
108                   elementBuilder.weight(weights.get(i));
109                rset.add(elementBuilder.build());
110             }
111             rsets.add(rset.build());
112          }
113          builder.recordSets(rsets.build());
114          return builder.build();
115       }
116    }

133    private static class GeoServiceTypeAdapter implements JsonDeserializer<GeoService> {
134 
135       @Override
136       public GeoService deserialize(JsonElement json, Type typeOfT, JsonDeserializationContext context)
137             throws JsonParseException {
138          CreepyGeoService creepyGeoService = context.deserialize(json, CreepyGeoService.class);
139          GeoService.Builder builder = GeoService.builder();
140          builder.name(creepyGeoService.name);
141          builder.active("Y".equals(creepyGeoService.active));
142          builder.ttl(creepyGeoService.ttl);
143          builder.nodes(creepyGeoService.nodes);
144          builder.groups(creepyGeoService.groups);
145          return builder.build();
146       }
