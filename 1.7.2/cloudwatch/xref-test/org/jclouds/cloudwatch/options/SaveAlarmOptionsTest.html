62     public void testPopulatedOptions() throws Exception {
63        boolean actionsEnabled = false;
64        Set<String> alarmActions = ImmutableSet.of(
65              "TestAlarmAction1",
66              "TestAlarmAction2"
67        );
68        String alarmDescription = "TestAlarmDescription";
69        String alarmName = "TestAlarmName";
70        ComparisonOperator comparisonOperator = ComparisonOperator.GREATER_THAN_OR_EQUAL_TO_THRESHOLD;
71        Set<Dimension> dimensions = ImmutableSet.of(
72              new Dimension("TestDimension1", "TestValue1"),
73              new Dimension("TestDimension2", "TestValue2")
74        );
75        int evaluationPeriods = 360;
76        Set<String> insufficientDataActions = ImmutableSet.of(
77              "TestInsufficientDataAction1",
78              "TestInsufficientDataAction2"
79        );
80        String metricName = "TestMetricName";
81        String namespace = Namespaces.AUTO_SCALING;
82        Set<String> okActions = ImmutableSet.of(
83              "TestOKAction1",
84              "TestOKAction2"
85        );
86        int period = 300;
87        Statistics statistic = Statistics.SAMPLE_COUNT;
88        double threshold = 1.0;
89        Unit unit = Unit.BITS;
90        Multimap<String, String> formParameters = new SaveAlarmOptions()
91              .actionsEnabled(actionsEnabled)
92              .alarmActions(alarmActions)
93              .alarmDescription(alarmDescription)
94              .alarmName(alarmName)
95              .comparisonOperator(comparisonOperator)
96              .dimensions(dimensions)
97              .evaluationPeriods(evaluationPeriods)
98              .insufficientDataActions(insufficientDataActions)
99              .metricName(metricName)
100             .namespace(namespace)
101             .okActions(okActions)
102             .period(period)
103             .statistic(statistic)
104             .threshold(threshold)
105             .unit(unit)
106             .buildFormParameters();
107       int alarmActionIndex = 1;
108       int dimensionIndex = 1;
109       int insufficientDataActionIndex = 1;
110       int okActionIndex = 1;
111 
112       for (String alarmAction : alarmActions) {
113          assertEquals(formParameters.get("AlarmActions.member." + alarmActionIndex), ImmutableSet.of(alarmAction));
114          alarmActionIndex++;
115       }
116 
117       for (Dimension dimension : dimensions) {
118          assertEquals(formParameters.get("Dimensions.member." + dimensionIndex + ".Name"),
119                       ImmutableSet.of(dimension.getName()));
120          assertEquals(formParameters.get("Dimensions.member." + dimensionIndex + ".Value"),
121                       ImmutableSet.of(dimension.getValue()));
122          dimensionIndex++;
123       }
124 
125       for (String insufficientDataAction : insufficientDataActions) {
126          assertEquals(formParameters.get("InsufficientDataActions.member." + insufficientDataActionIndex),
127                       ImmutableSet.of(insufficientDataAction));
128          insufficientDataActionIndex++;
129       }
130 
131       for (String okAction : okActions) {
132          assertEquals(formParameters.get("OKActions.member." + okActionIndex), ImmutableSet.of(okAction));
133          okActionIndex++;
134       }
135 
136       assertEquals(formParameters.get("ActionsEnabled"), ImmutableSet.of(Boolean.toString(actionsEnabled)));
137       assertEquals(formParameters.get("AlarmDescription"), ImmutableSet.of(alarmDescription));
138       assertEquals(formParameters.get("AlarmName"), ImmutableSet.of(alarmName));
139       assertEquals(formParameters.get("ComparisonOperator"), ImmutableSet.of(comparisonOperator.toString()));
140       assertEquals(formParameters.get("EvaluationPeriods"), ImmutableSet.of(Integer.toString(evaluationPeriods)));
141       assertEquals(formParameters.get("MetricName"), ImmutableSet.of(metricName));
142       assertEquals(formParameters.get("Namespace"), ImmutableSet.of(namespace));
143       assertEquals(formParameters.get("Period"), ImmutableSet.of(Integer.toString(period)));
144       assertEquals(formParameters.get("Statistic"), ImmutableSet.of(statistic.toString()));
145       assertEquals(formParameters.get("Threshold"), ImmutableSet.of(Double.toString(threshold)));
146       assertEquals(formParameters.get("Unit"), ImmutableSet.of(unit.toString()));
147    }
