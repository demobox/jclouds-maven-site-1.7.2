354    private HttpRequest alarmRequest(Map<String, String> arguments) {
355       Map<String, String> sortedArguments = Maps.newTreeMap();
356       Map<String, String> defaultArguments =
357             ImmutableMap.<String, String> builder()
358                         .put("SignatureMethod", "HmacSHA256")
359                         .put("SignatureVersion", "2")
360                         .put("Timestamp", "2009-11-08T15%3A54%3A08.897Z")
361                         .put("Version", "2010-08-01")
362                         .build();
363 
364       sortedArguments.putAll(arguments);
365 
366       for (Map.Entry<String, String> defaultArgument : defaultArguments.entrySet()) {
367          if (!sortedArguments.containsKey(defaultArgument.getKey())) {
368             sortedArguments.put(defaultArgument.getKey(), defaultArgument.getValue());
369          }
370       }
371 
372       Map<String, String> realArguments = Maps.newLinkedHashMap(sortedArguments);
373 
374       realArguments.put("AWSAccessKeyId", "identity");
375 
376       return HttpRequest.builder()
377                         .method("POST")
378                         .endpoint("https://monitoring.us-east-1.amazonaws.com/")
379                         .addHeader("Host", "monitoring.us-east-1.amazonaws.com")
380                         .payload(payloadFromStringWithContentType(Joiner.on("&")
381                                                                         .withKeyValueSeparator("=")
382                                                                         .join(realArguments).trim(),
383                                                                   "application/x-www-form-urlencoded"))
384                         .build();
385    }
