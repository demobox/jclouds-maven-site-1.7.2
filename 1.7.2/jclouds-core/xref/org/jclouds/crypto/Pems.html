78     private static class PemProcessor<T> implements ByteProcessor<T> {
79        private interface ResultParser<T> {
80           T parseResult(byte[] bytes) throws IOException;
81        }
82  
83        private final ByteArrayOutputStream out = new ByteArrayOutputStream();
84        private final Map<String, ResultParser<T>> parsers;
85  
86        private PemProcessor(Map<String, ResultParser<T>> parsers) {
87           this.parsers = checkNotNull(parsers, "parsers");
88        }
89  
90        @Override
91        public boolean processBytes(byte[] buf, int off, int len) {
92           out.write(buf, off, len);
93           return true;
94        }
95  
96        @Override
97        public T getResult() {
98           Pem pem = PemReader.INSTANCE.apply(new String(out.toByteArray(), US_ASCII));
99           String beginMarker = "-----BEGIN " + pem.type + "-----";
100          checkState(parsers.containsKey(beginMarker), "Invalid PEM: no parsers for marker %s in %s", beginMarker,
101                parsers.keySet());
102          try {
103             return parsers.get(beginMarker).parseResult(pem.content);
104          } catch (IOException e) {
105             throw new IllegalStateException("Invalid PEM : " + pem, e);
106          }
107       }
108    }

110    /**
111     * Parsed PEM format
112     * 
113     * <pre>
114     *  -----BEGIN RSA PRIVATE KEY-----
115     *  Proc-Type: 4,ENCRYPTED
116     *  DEK-Info: DES-EDE3-CBC,3F17F5316E2BAC89
117     * 
118     *  ...base64 encoded data...
119     *  -----END RSA PRIVATE KEY-----
120     * 
121     * </pre>
122     * 
123     */
124    private static enum PemReader implements Function<CharSequence, Pem> {
125       INSTANCE;
126       private static final String BEGIN = "-----BEGIN ";
127       private static final String END = "-----END ";
128 
129       @Override
130       public Pem apply(CharSequence chars) {
131          checkNotNull(chars, "chars");
132          BufferedReader reader = null;
133          try {
134            reader  =  new BufferedReader(new StringReader(chars.toString()));
135             Optional<String> begin = skipUntilBegin(reader);
136             checkArgument(begin.isPresent(), "chars %s doesn't contain % line", chars, BEGIN);
137             String line = begin.get().substring(BEGIN.length());
138             String type = line.substring(0, line.indexOf('-'));
139             StringBuilder encoded = new StringBuilder();
140 
141             boolean reachedEnd = false;
142             while ((line = reader.readLine()) != null) {
143                if (line.indexOf(':') >= 0) { // skip headers
144                   continue;
145                }
146                if (line.indexOf(END + type) != -1) {
147                   reachedEnd = true;
148                   break;
149                }
150                encoded.append(line.trim());
151             }
152 
153             checkArgument(reachedEnd, "chars %s doesn't contain % line", chars, END);
154             return new Pem(type, base64().decode(encoded.toString()));
155          } catch (IOException e) {
156             throw new IllegalStateException(String.format("io exception reading %s", chars), e);
157          } finally {
158             closeQuietly(reader);
159          }
160 
161       }
162 
163       private static Optional<String> skipUntilBegin(BufferedReader reader) throws IOException {
164          String line = reader.readLine();
165          while (line != null && !line.startsWith(BEGIN)) {
166             line = reader.readLine();
167          }
168          return Optional.fromNullable(line);
169       }
170    }

184    /**
185     * Returns the object of generic type {@code T} that is pem encoded in the supplier.
186     * 
187     * @param supplier
188     *           the input stream factory
189     * @param marker
190     *           header that begins the PEM block
191     * @param processor
192     *           how to parser the object from a byte array
193     * @return the object of generic type {@code T} which was PEM encoded in the stream
194     * @throws IOException
195     *            if an I/O error occurs
196     */
197    public static <T> T fromPem(InputSupplier<? extends InputStream> supplier, PemProcessor<T> processor)
198          throws IOException {
199       try {
200          return readBytes(supplier, processor);
201       } catch (RuntimeException e) {
202          propagateIfInstanceOf(e.getCause(), IOException.class);
203          propagateIfInstanceOf(e, IOException.class);
204          throw e;
205       }
206    }

322     * Returns the {@link X509EncodedKeySpec} that is pem encoded in the supplier.
323     * 
324     * @param supplier
325     *           the input stream factory
326     * @param certFactory
327     *           or null to use default
328     *

332     * @throws CertificateException
333     */
334    public static X509Certificate x509Certificate(InputSupplier<? extends InputStream> supplier,
335          @Nullable CertificateFactory certFactory) throws IOException, CertificateException {
336       final CertificateFactory certs = certFactory != null ? certFactory : CertificateFactory.getInstance("X.509");
337       try {
338          return fromPem(
339                supplier,
340                new PemProcessor<X509Certificate>(ImmutableMap.<String, PemProcessor.ResultParser<X509Certificate>> of(
341                      CERTIFICATE_X509_MARKER, new PemProcessor.ResultParser<X509Certificate>() {
342 
343                         @Override
344                         public X509Certificate parseResult(byte[] bytes) throws IOException {
345                            try {
346                               return (X509Certificate) certs.generateCertificate(new ByteArrayInputStream(bytes));
347                            } catch (CertificateException e) {
348                               throw new RuntimeException(e);
349                            }
350                         }
351 
352                      })));
353       } catch (RuntimeException e) {
354          propagateIfInstanceOf(e.getCause(), CertificateException.class);
355          throw e;
356       }
357    }

394    /**

398       String marker = key instanceof RSAPrivateCrtKey ? PRIVATE_PKCS1_MARKER : PRIVATE_PKCS8_MARKER;
399       return pem(key instanceof RSAPrivateCrtKey ? encode(RSAPrivateCrtKey.class.cast(key)) : key.getEncoded(), marker);

402    private static String pem(byte[] encoded, String marker) {
403       String ls = System.getProperty("line.separator");
404       StringBuilder builder = new StringBuilder();
405       builder.append(marker).append(ls);
406       builder.append(on(ls).join(fixedLength(64).split(base64().encode(encoded)))).append(ls);
407       builder.append(marker.replace("BEGIN", "END")).append(ls);
408       return builder.toString();
409    }
