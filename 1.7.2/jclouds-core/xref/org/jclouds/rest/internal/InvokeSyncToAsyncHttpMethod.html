70     @Inject
71     @VisibleForTesting
72     InvokeSyncToAsyncHttpMethod(Function<Invocation, Invocation> sync2async, Function<Invocation, HttpRequest> annotationProcessor,
73           HttpCommandExecutorService http, Function<HttpRequest, Function<HttpResponse, ?>> transformerForRequest,
74           TimeLimiter timeLimiter, InvocationConfig config,
75           @Named(PROPERTY_USER_THREADS) ListeningExecutorService userExecutor) {
76        this.sync2async = sync2async;
77        this.annotationProcessor = annotationProcessor;
78        this.http = http;
79        this.userExecutor = userExecutor;
80        this.timeLimiter = timeLimiter;
81        this.transformerForRequest = transformerForRequest;
82        this.config = config;
83     }

85     @Override
86     public Object apply(Invocation in) {
87        if (isFuture(in.getInvokable())) {
88           return submit(in);
89        }
90        Invocation async = toAsync(in);
91        Optional<Long> timeoutNanos = config.getTimeoutNanos(async);
92        if (timeoutNanos.isPresent()) {
93           return invokeWithTimeout(async, timeoutNanos.get());
94        }
95        return invoke(async);
96     }

98     /**
99      * submits the {@linkplain HttpCommand} associated with {@code invocation},
100     * {@link #getTransformer(String, HttpCommand) parses its response}, and
101     * applies a {@link #getFallback(String, Invocation, HttpCommand) fallback}
102     * if a {@code Throwable} is encountered. Parsing and Fallback occur on the
103     * {@code userExecutor} thread.
104     */
105    public ListenableFuture<?> submit(Invocation invocation) {
106       String commandName = config.getCommandName(invocation);
107       HttpCommand command = toCommand(commandName, invocation);
108       Function<HttpResponse, ?> transformer = getTransformer(commandName, command);
109       org.jclouds.Fallback<?> fallback = getFallback(commandName, invocation, command);
110 
111       logger.debug(">> submitting %s", commandName);
112       return withFallback(transform(http.submit(command), transformer, userExecutor), fallback);
113    }

115    /**
116     * invokes the {@linkplain HttpCommand} associated with {@code invocation},
117     * {@link #getTransformer(String, HttpCommand) parses its response}, and
118     * applies a {@link #getFallback(String, Invocation, HttpCommand) fallback}
119     * if a {@code Throwable} is encountered.
120     */
121    public Object invoke(Invocation invocation) {
122       String commandName = config.getCommandName(invocation);
123       HttpCommand command = toCommand(commandName, invocation);
124       Function<HttpResponse, ?> transformer = getTransformer(commandName, command);
125       org.jclouds.Fallback<?> fallback = getFallback(commandName, invocation, command);
126 
127       logger.debug(">> invoking %s", commandName);
128       try {
129          return transformer.apply(http.invoke(command));
130       } catch (Throwable t) {
131          try {
132             return fallback.createOrPropagate(t);
133          } catch (Exception e) {
134             throw propagate(e);
135          }
136       }
137    }

139    /**
140     * calls {@link #invoke(Invocation)}, timing out after the specified time
141     * limit. If the target method call finished before the limit is reached, the
142     * return value or exception is propagated to the caller exactly as-is. If,
143     * on the other hand, the time limit is reached, we attempt to abort the call
144     * to the target, and throw an {@link UncheckedTimeoutException} to the
145     * caller.
146     * 
147     * @param invocation
148     *           the Invocation to invoke via {@link #invoke(Invocation)}
149     * @param limitNanos
150     *           with timeoutUnit, the maximum length of time to wait in
151     *           nanoseconds
152     * @throws InterruptedException
153     *            if our thread is interrupted during execution
154     * @throws UncheckedTimeoutException
155     *            if the time limit is reached
156     * @see TimeLimiter#callWithTimeout(Callable, long, TimeUnit, boolean)
157     */
158    public Object invokeWithTimeout(final Invocation invocation, final long limitNanos) {
159       String commandName = config.getCommandName(invocation);
160       HttpCommand command = toCommand(commandName, invocation);
161       org.jclouds.Fallback<?> fallback = getFallback(commandName, invocation, command);
162 
163       logger.debug(">> blocking on %s for %s", invocation, limitNanos);
164       try {
165          return timeLimiter
166                .callWithTimeout(new InvokeAndTransform(commandName, command), limitNanos, NANOSECONDS, true);
167       } catch (Throwable t) {
168          try {
169             return fallback.createOrPropagate(t);
170          } catch (Exception e) {
171             throw propagate(e);
172          }
173       }
174    }
175 
176    private org.jclouds.Fallback<?> getFallback(String commandName, Invocation invocation, HttpCommand command) {
177       HttpRequest request = command.getCurrentRequest();
178       org.jclouds.Fallback<?> fallback = config.getFallback(invocation);
179       if (fallback instanceof InvocationContext)
180          InvocationContext.class.cast(fallback).setContext(request);
181       logger.trace("<< exceptions from %s are parsed by %s", commandName, fallback.getClass().getSimpleName());
182       return fallback;
183    }

185    @VisibleForTesting
186    final class InvokeAndTransform implements Callable<Object> {
187       private final String commandName;
188       private final HttpCommand command;
189       private final Function<HttpResponse, ?> transformer;
190 
191       InvokeAndTransform(String commandName, HttpCommand command) {
192          this.commandName = commandName;
193          this.command = command;
194          this.transformer = getTransformer(commandName, command);
195       }
196 
197       @Override
198       public Object call() throws Exception {
199          return transformer.apply(http.invoke(command));
200       }
201 
202       @Override
203       public int hashCode() {
204          return Objects.hashCode(commandName, command, transformer);
205       }
206 
207       @Override
208       public boolean equals(Object obj) {
209          if (this == obj)
210             return true;
211          if (obj == null || getClass() != obj.getClass())
212             return false;
213          InvokeAndTransform that = InvokeAndTransform.class.cast(obj);
214          return equal(this.commandName, that.commandName) && equal(this.command, that.command)
215                && equal(this.transformer, that.transformer);
216       }
217 
218       @Override
219       public String toString() {
220          return toStringHelper(this).add("commandName", commandName).add("command", command)
221                .add("transformer", transformer).toString();
222       }
223    }

225    /**
226     * looks up the corresponding {@code Invocation} that returns a
227     * {@code Future}. Only Invokables that return {@code Futures} are annotated
228     * in a way that can be parsed into an {@linkplain HttpRequest}.
229     */
230    private Invocation toAsync(Invocation in) {
231       Invocation async = sync2async.apply(in);
232       checkState(isFuture(async.getInvokable()), "not a future: %s", async);
233       return async;
234    }
235 
236    private HttpCommand toCommand(String commandName, Invocation invocation) {
237       logger.trace(">> converting %s", commandName);
238       HttpRequest request = annotationProcessor.apply(invocation);
239       logger.trace("<< converted %s to %s", commandName, request.getRequestLine());
240       return new HttpCommand(request);
241    }
242 
243    private Function<HttpResponse, ?> getTransformer(String commandName, HttpCommand command) {
244       HttpRequest request = command.getCurrentRequest();
245       Function<HttpResponse, ?> transformer = transformerForRequest.apply(request);
246       logger.trace("<< response from %s is parsed by %s", commandName, transformer.getClass().getSimpleName());
247       return transformer;
248    }

250    private boolean isFuture(Invokable<?, ?> in) {
251       return in.getReturnType().getRawType().equals(ListenableFuture.class);
252    }
