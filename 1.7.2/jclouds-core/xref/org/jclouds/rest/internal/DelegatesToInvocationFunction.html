71   * @param <S>
72   *           The enclosing type of the interface that a dynamic proxy like this implements
73   * @param <F>
74   *           The function that implements this dynamic proxy

76  @Beta
77  public class DelegatesToInvocationFunction<S, F extends Function<Invocation, Object>> implements
78        InvocationHandler {

80     private static final Object[] NO_ARGS = {};
81  
82     /**
83      * {@inheritDoc}
84      * 
85      * <p>
86      * <ul>
87      * <li>{@code proxy.hashCode()} delegates to {@link AbstractInvocationHandler#hashCode}
88      * <li>{@code proxy.toString()} delegates to {@link AbstractInvocationHandler#toString}
89      * <li>{@code proxy.equals(argument)} returns true if:
90      * <ul>
91      * <li>{@code proxy} and {@code argument} are of the same type
92      * <li>and {@link AbstractInvocationHandler#equals} returns true for the {@link InvocationHandler} of
93      * {@code argument}
94      * </ul>
95      * <li>other method calls are dispatched to {@link #handleInvocation}.
96      * </ul>
97      * 
98      * @throws Throwable
99      */
100    @Override
101    public final Object invoke(Object proxy, Method invoked, @Nullable Object[] argv) throws Throwable {
102       if (argv == null) {
103          argv = NO_ARGS;
104       }
105       if (argv.length == 0 && invoked.getName().equals("hashCode")) {
106          return hashCode();
107       }
108       if (argv.length == 1 && invoked.getName().equals("equals") && invoked.getParameterTypes()[0] == Object.class) {
109          Object arg = argv[0];
110          return proxy.getClass().isInstance(arg) && equals(Proxy.getInvocationHandler(arg));
111       }
112       if (argv.length == 0 && invoked.getName().equals("toString")) {
113          return toString();
114       }
115       List<Object> args = Arrays.asList(argv);
116       if (all(args, notNull()))
117          args = ImmutableList.copyOf(args);
118       else
119          args = Collections.unmodifiableList(args);
120       Invokable<?, Object> invokable = method(ownerType, invoked);
121       Invocation invocation = Invocation.create(invokable, args);
122       try {
123          return handle(invocation);
124       } catch (Throwable e) {
125          propagateIfPossible(e, invocation.getInvokable().getExceptionTypes());
126          throw e;
127       }
128    }

130    private static final Invokable<?, ?> CLOSE;
131 
132    static {
133       try {
134          CLOSE = Invokable.from(Closeable.class.getMethod("close"));
135       } catch (SecurityException e) {
136          throw propagate(e);
137       } catch (NoSuchMethodException e) {
138          throw propagate(e);
139       }
140    }

142    protected Object handle(Invocation invocation) {
143       Invokable<?, ?> invokable = invocation.getInvokable();
144       if (isCloseMethod(invokable)) {
145          try {
146             injector.getInstance(Closer.class).close();
147             return null;
148          } catch (Throwable e) {
149             throw propagate(e);
150          }
151       } else if (invokable.isAnnotationPresent(Provides.class)) {
152          return lookupValueFromGuice(invokable);
153       } else if (invokable.isAnnotationPresent(Delegate.class)) {
154          return propagateContextToDelegate(invocation);
155       } else {
156          return methodInvoker.apply(invocation);
157       }
158    }
159 
160    private static boolean isCloseMethod(Invokable<?, ?> invokable) {
161       /*
162        * Tests equality according to the Javadoc for java.lang.reflect.Method:
163        *
164        * Two Methods are the same if they were declared by the same class
165        * and have the same name and formal parameter types and return type.
166        *
167        * Invokable now uses the *owning* class (not the declaring class) in
168        * its equals check.
169        */
170       return CLOSE.getDeclaringClass().equals(invokable.getDeclaringClass())
171               && CLOSE.getName().equals(invokable.getName())
172               && CLOSE.getParameters().equals(invokable.getParameters())
173               && CLOSE.getReturnType().equals(invokable.getReturnType());
174    }

192    private Object propagateContextToDelegate(Invocation caller) {
193       Class<?> returnType = unwrapIfOptional(caller.getInvokable().getReturnType());
194       Function<Invocation, Object> delegate;
195       setCaller.enter(caller);
196       try {
197          @SuppressWarnings("unchecked")
198          Key<Function<Invocation, Object>> delegateType = (Key<Function<Invocation, Object>>) methodInvokerFor(returnType);
199          delegate = injector.getInstance(delegateType);
200       } finally {
201          setCaller.exit();
202       }
203       Object result = FunctionalReflection.newProxy(returnType, delegate);
204       if (isReturnTypeOptional(caller.getInvokable())) {
205          result = optionalConverter.apply(InvocationSuccess.create(caller, result));
206       }
207       return result;
208    }

210    /**
211     * attempts to guess the generic type params for the delegate's invocation function based on the supplied type
212     */
213    protected Key<?> methodInvokerFor(Class<?> returnType) {
214       switch (methodInvoker.getClass().getTypeParameters().length) {
215       case 0:
216          return Key.get(methodInvoker.getClass());
217       case 1:
218          return Key.get(Types.newParameterizedType(methodInvoker.getClass(), returnType));
219       }
220       throw new IllegalArgumentException(returnType + " has too many type parameters");
221    }

223    static final Predicate<Annotation> isQualifierPresent = new Predicate<Annotation>() {
224       public boolean apply(Annotation input) {
225          return input.annotationType().isAnnotationPresent(Qualifier.class);
226       }
227    };

229    private Object lookupValueFromGuice(Invokable<?, ?> invoked) {
230       try {
231          Type genericReturnType = invoked.getReturnType().getType();
232          try {
233             Annotation qualifier = find(ImmutableList.copyOf(invoked.getAnnotations()), isQualifierPresent);
234             return getInstanceOfTypeWithQualifier(genericReturnType, qualifier);
235          } catch (ProvisionException e) {
236             throw propagate(e.getCause());
237          } catch (RuntimeException e) {
238             return instanceOfTypeOrPropagate(genericReturnType, e);
239          }
240       } catch (ProvisionException e) {
241          AuthorizationException aex = getFirstThrowableOfType(e, AuthorizationException.class);
242          if (aex != null)
243             throw aex;
244          throw e;
245       }
246    }
247 
248    Object instanceOfTypeOrPropagate(Type genericReturnType, RuntimeException e) {
249       try {
250          // look for an existing binding
251          Binding<?> binding = injector.getExistingBinding(Key.get(genericReturnType));
252          if (binding != null)
253             return binding.getProvider().get();
254 
255          // then, try looking via supplier
256          binding = injector.getExistingBinding(Key.get(newParameterizedType(Supplier.class, genericReturnType)));
257          if (binding != null)
258             return Supplier.class.cast(binding.getProvider().get()).get();
259 
260          // else try to create an instance
261          return injector.getInstance(Key.get(genericReturnType));
262       } catch (ConfigurationException ce) {
263          throw e;
264       }
265    }
266 
267    Object getInstanceOfTypeWithQualifier(Type genericReturnType, Annotation qualifier) {
268       // look for an existing binding
269       Binding<?> binding = injector.getExistingBinding(Key.get(genericReturnType, qualifier));
270       if (binding != null)
271          return binding.getProvider().get();
272 
273       // then, try looking via supplier
274       binding = injector
275             .getExistingBinding(Key.get(newParameterizedType(Supplier.class, genericReturnType), qualifier));
276       if (binding != null)
277          return Supplier.class.cast(binding.getProvider().get()).get();
278 
279       // else try to create an instance
280       return injector.getInstance(Key.get(genericReturnType, qualifier));
281    }
