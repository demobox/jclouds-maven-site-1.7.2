59  @Test(groups = "unit", singleThreaded = true)
60  public class InvokeMappedHttpMethodTest {
61  
62     public interface ThingApi {
63        HttpResponse get();
64     }
65  
66     public interface ThingAsyncApi {
67        @Named("ns:get")
68        ListenableFuture<HttpResponse> get();
69     }
70  
71     private Invocation get;
72     private Invocation asyncGet;
73     private Function<Invocation, Invocation> sync2async;
74     private HttpRequest getRequest = HttpRequest.builder().method("GET").endpoint("http://get").build();
75     private HttpCommand getCommand = new HttpCommand(getRequest);
76     private Function<Invocation, HttpRequest> toRequest;
77  
78     @BeforeClass
79     void setupInvocations() throws SecurityException, NoSuchMethodException {
80        get = Invocation.create(method(ThingApi.class, "get"), ImmutableList.of());
81        asyncGet = Invocation.create(method(ThingAsyncApi.class, "get"), ImmutableList.of());
82        sync2async = Functions.forMap(ImmutableMap.of(get, asyncGet));
83        toRequest = Functions.forMap(ImmutableMap.of(asyncGet, getRequest));
84     }
85  
86     @SuppressWarnings("unchecked")
87     private Function<HttpRequest, Function<HttpResponse, ?>> transformerForRequest = Function.class.cast(Functions
88           .constant(Functions.identity()));
89     private ListeningExecutorService userThreads = MoreExecutors.sameThreadExecutor();
90  
91     private HttpResponse response = HttpResponse.builder().statusCode(200).payload("foo").build();
92     private HttpCommandExecutorService http;
93     private TimeLimiter timeLimiter;
94     @SuppressWarnings("rawtypes")
95     private org.jclouds.Fallback fallback;
96     private InvocationConfig config;
97     private InvokeSyncToAsyncHttpMethod invokeHttpMethod;
98  
99     private ListenableFuture<HttpResponse> future;
100 
101    @SuppressWarnings("unchecked")
102    @BeforeMethod
103    void createMocks() {
104       http = createMock(HttpCommandExecutorService.class);
105       timeLimiter = createMock(TimeLimiter.class);
106       fallback = createMock(org.jclouds.Fallback.class);
107       config = createMock(InvocationConfig.class);
108       future = createMock(ListenableFuture.class);
109       invokeHttpMethod = new InvokeSyncToAsyncHttpMethod(sync2async, toRequest, http, transformerForRequest, timeLimiter, config,
110             userThreads);
111       expect(config.getCommandName(asyncGet)).andReturn("ns:get");
112       expect(config.getFallback(asyncGet)).andReturn(fallback);
113    }

115    @AfterMethod
116    void verifyMocks() {
117       verify(http, timeLimiter, fallback, config, future);
118    }

120    public void testMethodWithTimeoutRunsTimeLimiter() throws Exception {
121       expect(config.getTimeoutNanos(asyncGet)).andReturn(Optional.of(250000000l));
122       InvokeAndTransform invoke = invokeHttpMethod.new InvokeAndTransform("ns:get", getCommand);
123       expect(timeLimiter.callWithTimeout(invoke, 250000000, TimeUnit.NANOSECONDS, true)).andReturn(response);
124       replay(http, timeLimiter, fallback, config, future);
125       invokeHttpMethod.apply(get);
126    }
127 
128    public void testMethodWithNoTimeoutCallGetDirectly() throws Exception {
129       expect(config.getTimeoutNanos(asyncGet)).andReturn(Optional.<Long> absent());
130       expect(http.invoke(new HttpCommand(getRequest))).andReturn(response);
131       replay(http, timeLimiter, fallback, config, future);
132       invokeHttpMethod.apply(get);
133    }
134 
135    public void testAsyncMethodSubmitsRequest() throws Exception {
136       expect(http.submit(new HttpCommand(getRequest))).andReturn(future);
137       future.addListener(anyObject(Runnable.class), eq(userThreads));
138       replay(http, timeLimiter, fallback, config, future);
139       invokeHttpMethod.apply(asyncGet);
140    }

142    private HttpResponse fallbackResponse = HttpResponse.builder().statusCode(200).payload("bar").build();
143 
144    public void testDirectCallRunsFallbackCreateOrPropagate() throws Exception {
145       IllegalStateException exception = new IllegalStateException();
146       expect(config.getTimeoutNanos(asyncGet)).andReturn(Optional.<Long> absent());
147       expect(http.invoke(new HttpCommand(getRequest))).andThrow(exception);
148       expect(fallback.createOrPropagate(exception)).andReturn(fallbackResponse);
149       replay(http, timeLimiter, fallback, config, future);
150       assertEquals(invokeHttpMethod.apply(get), fallbackResponse);
151    }
152 
153    public void testTimeLimitedRunsFallbackCreateOrPropagate() throws Exception {
154       IllegalStateException exception = new IllegalStateException();
155       expect(config.getTimeoutNanos(asyncGet)).andReturn(Optional.of(250000000l));
156       InvokeAndTransform invoke = invokeHttpMethod.new InvokeAndTransform("ns:get", getCommand);
157       expect(timeLimiter.callWithTimeout(invoke, 250000000, TimeUnit.NANOSECONDS, true)).andThrow(exception);
158       expect(fallback.createOrPropagate(exception)).andReturn(fallbackResponse);
159       replay(http, timeLimiter, fallback, config, future);
160       assertEquals(invokeHttpMethod.apply(get), fallbackResponse);
161    }

163    @SuppressWarnings("unchecked")
164    public void testSubmitRunsFallbackCreateOnGet() throws Exception {
165       IllegalStateException exception = new IllegalStateException();
166       expect(http.submit(new HttpCommand(getRequest))).andReturn(
167             Futures.<HttpResponse> immediateFailedFuture(exception));
168       expect(fallback.create(exception)).andReturn(Futures.<HttpResponse> immediateFuture(fallbackResponse));
169       // not using the field, as you can see above we are making an immediate
170       // failed future instead.
171       future = createMock(ListenableFuture.class);
172       replay(http, timeLimiter, fallback, config, future);
173       assertEquals(ListenableFuture.class.cast(invokeHttpMethod.apply(asyncGet)).get(), fallbackResponse);
174    }
