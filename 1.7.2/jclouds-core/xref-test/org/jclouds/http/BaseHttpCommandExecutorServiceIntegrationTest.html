251    private static class MD5CheckDispatcher extends Dispatcher {
252 
253       @Override
254       public MockResponse dispatch(RecordedRequest request) throws InterruptedException {
255          try {
256             MockResponse response = new MockResponse();
257             String expectedMd5 = request.getHeader("Content-MD5");
258             ByteSource body = ByteSource.wrap(request.getBody());
259             String realMd5FromRequest = base64().encode(body.hash(md5()).asBytes());
260             boolean matched = expectedMd5.equals(realMd5FromRequest);
261             if (matched) {
262                response.addHeader("x-Content-MD5", realMd5FromRequest);
263             } else {
264                response.setResponseCode(500);
265             }
266             return response;
267          } catch (IOException ex) {
268             throw Throwables.propagate(ex);
269          }
270       }

274    @Test
275    public void testUploadBigFile() throws Exception {
276       MockWebServer server = mockWebServer(new MD5CheckDispatcher());
277       IntegrationTestClient client = client(server.getUrl("/").toString());
278 
279       File f = null;
280       Payload payload = null;
281 
282       try {
283          f = File.createTempFile("jclouds", "tmp");
284          f.deleteOnExit();
285          long length = (new Random().nextInt(32) + 1) * 1024 * 1024;
286 
287          MessageDigest digester = md5Digest();
288 
289          CharSink fileSink = Files.asCharSink(f, Charsets.UTF_8);
290          Writer out = null;
291          try {
292             out = fileSink.openStream();
293             for (long i = 0; i < length; i++) {
294                out.append('a');
295                digester.update((byte) 'a');
296             }
297             out.flush();
298          } finally {
299             close(out, true);
300          }
301 
302          payload = newByteSourcePayload(asByteSource(f));
303          byte[] digest = digester.digest();
304          String strDigest = base64().encode(digest);
305 
306          payload.getContentMetadata().setContentMD5(digest);
307          payload.getContentMetadata().setContentLength(f.length());
308          Multimap<String, String> headers = client.postPayloadAndReturnHeaders("", payload);
309          RecordedRequest request = server.takeRequest();
310          assertEquals(request.getHeader("Content-MD5"), strDigest);
311          assertEquals(headers.get("x-Content-MD5"), ImmutableList.of(strDigest));
312       } finally {
313          if (payload != null) {
314             payload.release();
315          }
316          if (f != null && f.exists()) {
317             f.delete();
318          }
319          close(client, true);
320          server.shutdown();
321       }
322    }
