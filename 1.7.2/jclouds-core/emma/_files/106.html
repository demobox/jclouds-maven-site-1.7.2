595      Set<Parameter> binderOrWrapWith = ImmutableSet.copyOf(concat(
596            parametersWithAnnotation(invocation.getInvokable(), BinderParam.class),
597            parametersWithAnnotation(invocation.getInvokable(), WrapWith.class)));
598      OUTER: for (Parameter entry : binderOrWrapWith) {

607         Object arg = args.size() >= position + 1 ? args.get(position) : null;
608         if (args.size() >= position + 1 && arg != null) {
609            Class<?> parameterType = entry.getType().getRawType();
610            Class<? extends Object> argType = arg.getClass();

613                                                                // {

615               if (arrayLength == 0)
616                  break OUTER;
617               arg = (Object[]) Array.newInstance(arg.getClass(), arrayLength);
618               System.arraycopy(args.toArray(), position, arg, 0, arrayLength);
619               shouldBreak = true;
620            } else if (argType.isArray() && parameterType.isArray()) {// TODO:
621                                                                      // &&
622                                                                      // invocation.getInvokable().isVarArgs())
623                                                                      // {
624            } else {
625               if (arg.getClass().isArray()) {

628               }
629            }

631               request = binder.bindToRequest(request, arg);
632            }
633            if (shouldBreak)
634               break OUTER;
635         } else {

642               continue OUTER;
643            }
644            checkNotNull(arg, invocation.getInvokable().getName() + " parameter " + (position + 1));

769   private Optional<?> getParamValue(Invocation invocation, @Nullable ParamParser extractor, int argIndex,
770         String paramKey) {
771      Object arg = invocation.getArgs().get(argIndex);
772      if (extractor != null && checkPresentOrNullable(invocation, paramKey, argIndex, arg)) {
773         // ParamParsers can deal with nullable parameters
774         arg = injector.getInstance(extractor.value()).apply(arg);
775      }
776      checkPresentOrNullable(invocation, paramKey, argIndex, arg);
777      return Optional.fromNullable(arg);
778   }

780   private boolean checkPresentOrNullable(Invocation invocation, String paramKey, int argIndex, Object arg) {
781      if (arg == null && !getInvokableParameters(invocation.getInvokable()).get(argIndex).isAnnotationPresent(Nullable.class))
782         throw new NullPointerException(format("param{%s} for invocation %s.%s", paramKey, invocation.getInvokable()
783               .getOwnerType().getRawType().getSimpleName(), invocation.getInvokable().getName()));
784      return true;
