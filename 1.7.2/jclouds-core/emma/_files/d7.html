386   public static Injector buildInjector(String name, ProviderMetadata providerMetadata, Supplier<Credentials> creds, List<Module> inputModules) {
387      List<Module> modules = newArrayList();
388      modules.addAll(inputModules);
389      boolean apiModuleSpecifiedByUser = apiModulePresent(inputModules);
390      Iterable<Module> defaultModules = ifSpecifiedByUserDontIncludeDefaultApiModule(
391               providerMetadata.getApiMetadata(), apiModuleSpecifiedByUser);
392      addAll(modules, defaultModules);
393      addClientModuleIfNotPresent(providerMetadata.getApiMetadata(), modules);
394      addRestContextBinding(providerMetadata.getApiMetadata(), modules);
395      addLoggingModuleIfNotPresent(modules);
396      addHttpModuleIfNeededAndNotPresent(modules);
397      addExecutorServiceIfNotPresent(modules);
398      addEventBusIfNotPresent(modules);
399      addCredentialStoreIfNotPresent(modules);
400      modules.add(new LifeCycleModule());
401      modules.add(new BindProviderMetadataContextAndCredentials(providerMetadata, creds));
402      modules.add(new BindNameToContext(name));
403      Injector returnVal = Guice.createInjector(Stage.PRODUCTION, modules);
404      returnVal.getInstance(ExecutionList.class).execute();
405      return returnVal;
406   }
