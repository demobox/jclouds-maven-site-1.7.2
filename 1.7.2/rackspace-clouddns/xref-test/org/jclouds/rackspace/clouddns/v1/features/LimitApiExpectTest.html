37  @Test(groups = "unit")
38  public class LimitApiExpectTest extends BaseCloudDNSApiExpectTest<CloudDNSApi> {
39     public void testListLimits() {
40        URI endpoint = URI.create("https://dns.api.rackspacecloud.com/v1.0/123123/limits");
41        LimitApi api = requestsSendResponses(
42              rackspaceAuthWithUsernameAndApiKey,
43              responseWithAccess, 
44              authenticatedGET().endpoint(endpoint).build(),
45              HttpResponse.builder().statusCode(OK.getStatusCode()).payload(payloadFromResource("/limit-list.json")).build()
46        ).getLimitApi();
47  
48        Limits limits = api.list();
49        assertEquals(limits.getAbsoluteLimits().get("records per domain"), Integer.valueOf(500));
50        assertEquals(limits.getAbsoluteLimits().get("domains"), Integer.valueOf(500));
51        assertEquals(Iterables.size(limits.getRateLimits()), 2);
52        
53        RateLimit statusRateLimit = Iterables.tryFind(limits.getRateLimits(), isStatusRateLimit()).orNull();
54        assertEquals(statusRateLimit.getRegex(), ".*/v\\d+\\.\\d+/(\\d+/status).*");
55        assertEquals(Iterables.get(statusRateLimit.getLimits(), 0).getVerb(), "GET");
56        assertEquals(Iterables.get(statusRateLimit.getLimits(), 0).getValue(), 5);
57        assertEquals(Iterables.get(statusRateLimit.getLimits(), 0).getRemaining().get(), Integer.valueOf(5));
58        assertEquals(Iterables.get(statusRateLimit.getLimits(), 0).getUnit(), "SECOND");
59     }
60     
61     public void testListLimitTypes() {
62        URI endpoint = URI.create("https://dns.api.rackspacecloud.com/v1.0/123123/limits/types");
63        LimitApi api = requestsSendResponses(
64              rackspaceAuthWithUsernameAndApiKey,
65              responseWithAccess, 
66              authenticatedGET().endpoint(endpoint).build(),
67              HttpResponse.builder().statusCode(OK.getStatusCode()).payload(payloadFromResource("/limit-types-list.json")).build()
68        ).getLimitApi();
69  
70        Iterable<String> limitTypes = api.listTypes();
71        assertEquals(Iterables.size(limitTypes), 3);
72     }
73     
74     private static Predicate<RateLimit> isStatusRateLimit() {
75        return new Predicate<RateLimit>() {
76           @Override
77           public boolean apply(RateLimit rateLimit) {
78              return rateLimit.getUri().contains("status");
79           }
80        };
81     }
