40  // NOTE:without testName, this will not call @Before* and fail w/NPE during

42  @Test(groups = "unit", testName = "OrgApiTest")
43  public class OrgApiTest extends BaseVCloudApiTest<OrgApi> {
44  
45     public void testlistOrgs() throws SecurityException, NoSuchMethodException, IOException {
46        Invokable<?, ?> method = method(OrgApi.class, "listOrgs");
47        GeneratedHttpRequest request = processor.createRequest(method, ImmutableList.of());
48  
49        assertRequestLineEquals(request, "GET https://vcenterprise.bluelock.com/api/v1.0/org HTTP/1.1");
50        assertNonPayloadHeadersEqual(request, "Accept: application/vnd.vmware.vcloud.orgList+xml\n");
51        assertPayloadEquals(request, null, null, false);
52  
53        assertResponseParserClassEquals(method, request, ParseSax.class);
54        assertSaxResponseParserClassEquals(method, OrgListHandler.class);
55        assertFallbackClassEquals(method, MapHttp4xxCodesToExceptions.class);
56  
57        checkFilters(request);
58     }
59  
60     public void testOrg() throws SecurityException, NoSuchMethodException, IOException {
61        Invokable<?, ?> method = method(OrgApi.class, "getOrg", URI.class);
62        GeneratedHttpRequest request = processor.createRequest(method, ImmutableList.<Object> of(URI
63                 .create("https://vcenterprise.bluelock.com/api/v1.0/org/1")));
64  
65        assertRequestLineEquals(request, "GET https://vcenterprise.bluelock.com/api/v1.0/org/1 HTTP/1.1");
66        assertNonPayloadHeadersEqual(request, "Accept: application/vnd.vmware.vcloud.org+xml\n");
67        assertPayloadEquals(request, null, null, false);
68  
69        assertResponseParserClassEquals(method, request, ParseSax.class);
70        assertSaxResponseParserClassEquals(method, OrgHandler.class);
71        assertFallbackClassEquals(method, NullOnNotFoundOr404.class);
72  
73        checkFilters(request);
74     }
